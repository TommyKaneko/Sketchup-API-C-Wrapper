44/**
45 * @brief Wrapper for SUOptionsProviderRef, providing access to a named
46 * set of key-value option pairs.
47 *
48 * An OptionsProvider is obtained from an OptionsManager and exposes
49 * the keys and typed values for a particular provider
50 * (e.g. "PageOptions", "UnitsOptions", etc.).
51 *
52 * @see OptionsManager
53 * @see https://extensions.sketchup.com/developer_center/sketchup_c_api/
54 */
55class OptionsProvider {
56 private:
57 SUOptionsProviderRef m_options_provider;
58 public:
59 /** @brief Default constructor – creates an invalid OptionsProvider. */
60 OptionsProvider();
61
62 /**
63 * @brief Constructs an OptionsProvider wrapping a raw API ref.
64 * @param ref the SUOptionsProviderRef to wrap.
65 */
66 OptionsProvider(SUOptionsProviderRef ref);
67
68 /**
69 * @brief Returns all key names for this provider.
70 * @return a vector of key name strings.
71 */
72 std::vector<std::string> keys() const;
73
74 /**
75 * @brief Returns the value for the given key.
76 * @param key the key name to look up.
77 * @return the TypedValue associated with the key.
78 */
79 TypedValue get_value(std::string key) const;
80
81 /**
82 * @brief Sets the value for the given key.
83 * @param key the key name to set.
84 * @param tval the TypedValue to assign.
85 * @return true on success, false on failure.
86 */
87 bool set_value(std::string key, const TypedValue& tval);
88};

90/**

97 * @see OptionsProvider
98 * @see https://extensions.sketchup.com/developer_center/sketchup_c_api/
99 */
100class OptionsManager {

103

106 OptionsManager();
107
108 /**
109 * @brief Constructs an OptionsManager wrapping a raw API ref.
110 * @param options_manager the SUOptionsManagerRef to wrap.
111 */
112 OptionsManager(SUOptionsManagerRef options_manager);
113
114 /**
115 * @brief Returns the names of all option providers.
116 * @return a vector of provider name strings.
117 */
118 std::vector<std::string> get_provider_names() const;
119
120 /**

122 * @param name the provider name (e.g. "PageOptions").
