34/**
35 * @brief Wrapper for SUColor, representing an RGBA color value.
36 *
37 * Provides direct access to red, green, blue, and alpha components
38 * via public reference members, and conversion to/from SUColor.
39 *
40 * @see https://extensions.sketchup.com/developer_center/sketchup_c_api/
41 */
42class Color {
43 private:
44 SUColor m_color;
45
46 public:
47 SUByte& red; ///< Red component (0-255).
48 SUByte& green; ///< Green component (0-255).
49 SUByte& blue; ///< Blue component (0-255).
50 SUByte& alpha; ///< Alpha component (0-255, where 255 is fully opaque).
51
52 /** @brief Default constructor – creates a black, fully opaque color. */
53 Color();
54
55 /**
56 * @brief Constructs a Color from a raw SUColor struct.
57 * @param color_input the SUColor to wrap.
58 */
59 Color(SUColor color_input);
60
61 /**
62 * @brief Returns the underlying SUColor struct.
63 * @return the raw SUColor value.
64 */
65 SUColor ref() const;
66
67 /** @brief Conversion operator to SUColor. */
68 operator SUColor() const;
69
70 /**
71 * @brief Compares two Color objects for equality.
72 * @return true if all RGBA components are equal.
73 */
74 friend bool operator ==(const Color &lhs, const Color &rhs);
75
76};
