43TEST_F(ModelLoad, DISABLED_EdgeReferencing)

46 ASSERT_FALSE(!m_model);
47
48 // Get all edges from the loaded model (include non-stray edges)
49 std::vector<Edge> edges = m_model->entities().edges(false);
50 // The above operation should refer to the same edges
51 size_t i = 0;
52 for (const Edge& edge : edges) {
53 if (i > 0) break; // only check one edge
54 i++;
55 CW::Edge edge_ref = edge;

123// Edge Explicit Copying - construct detached edges and add them to a new model
124TEST_F(ModelLoad, DISABLED_EdgeExplicitCopying)
125{
126 using namespace CW;
127 ASSERT_FALSE(!m_model);
128
129 std::vector<Edge> edges = m_model->entities().edges(false);
130
131 // Make explicit detached copies
132 size_t i = 0;
133 std::vector<Edge> edges_copy; edges_copy.reserve(edges.size());
134 std::cerr << "\033[33m[ INFO ] " << "NOTE: Materials and Layers cannot be assigned to Edges that are not attached to the model. EdgeExplicitCopying Test is expected to fail for equality of material and layer input and output. \033[0m" << std::endl;
135 for (const Edge& edge : edges) {
136 // if (i > 0) break; // only check one edge
137 // i++;
138 CW::Edge copy = edge.copy();
139 // copy properties
140 copy.color(edge.color());
141 copy.smooth(edge.smooth());
142 copy.soft(edge.soft());
143
144 ASSERT_FALSE(copy.attached());
145 EXPECT_NE(copy.ref().ptr, edge.ref().ptr);
146 ASSERT_NO_FATAL_FAILURE(EdgesAreEqual(copy, edge));
147 edges_copy.push_back(copy);
148 }
149
150 // Add detached copies into new model
151 CW::Entities dest_entities = m_model_copy->entities();
152 dest_entities.add_edges(edges_copy);
153 i = 0;
154 std::vector<Edge> copied_edges = dest_entities.edges(false);
155 EXPECT_EQ(edges.size(), copied_edges.size());
156 for (const Edge& original : edges) {
157 // if (i > 0) break; // only check one edge
158 // i++;
159 // find matching copied edge by start/end coords
160 auto found_it = std::find_if(copied_edges.begin(), copied_edges.end(),
161 [&original](const Edge& value) {
162 return (original.start().position() == value.start().position())
163 && (original.end().position() == value.end().position());
164 });
165 if (found_it != copied_edges.end()) {
166 const Edge& copied = *found_it;
167 EXPECT_NE(original.ref().ptr, copied.ref().ptr);
168 EXPECT_TRUE(copied.attached());
169 ASSERT_NO_FATAL_FAILURE(EdgesAreEqual(original, copied));
170 }
171 }
172}

181TEST_F(EdgeModelTest, CopyEdges) {
182 using namespace CW;
183 Entities src_entities = m_model->entities();
184 std::vector<Edge> source_edges = src_entities.edges(false);
185 ASSERT_GT(source_edges.size(), 0u);
186
187 // Create detached copies of each edge
188 std::vector<Edge> new_edges;
189 for (const auto& src : source_edges) {
190 new_edges.push_back(src.copy());
191 }
192
193 Entities dest_entities = m_model_copy->entities();
194 dest_entities.add_edges(new_edges);
195
196 // Read back from destination and verify count matches
197 std::vector<Edge> copied_edges = dest_entities.edges(false);
198 ASSERT_EQ(copied_edges.size(), source_edges.size());
199
200 // Verify start/end vertex positions match
201 for (size_t i = 0; i < source_edges.size(); ++i) {
202 Point3D src_start = source_edges[i].start().position();
203 Point3D src_end = source_edges[i].end().position();
204 // Find matching edge in copied_edges by vertex positions
205 auto found = std::find_if(copied_edges.begin(), copied_edges.end(),
206 [&src_start, &src_end](const Edge& e) {
207 return (e.start().position() == src_start && e.end().position() == src_end);
208 });
209 EXPECT_NE(found, copied_edges.end())
210 << "Could not find matching edge for source edge " << i;
211 }
212
213 SaveModel("Edges");
214}
