56 std::vector<Face> Entities::faces() const {

58  throw std::logic_error("CW::Entities::faces(): Entities is null");
59  }
60  size_t count = 0;
61  SUResult res = SUEntitiesGetNumFaces(m_entities, &count);
62  assert(res == SU_ERROR_NONE);
63  if (count == 0) {
64  return std::vector<Face>(0);
65  }
66  std::vector<SUFaceRef> face_refs(count, SU_INVALID);
67  res = SUEntitiesGetFaces(m_entities, count, face_refs.data(), &count);
68  assert(res == SU_ERROR_NONE);
69  std::vector<Face> faces(count);
70  std::transform(face_refs.begin(), face_refs.end(), faces.begin(),
71  [](const SUFaceRef& value) {
72  return Face(value);
73  });
74  return faces;

365  // Add geometry one by one to Geometry input.
