50SUCameraRef Camera::copy_reference(const Camera& other) {
51 if (other.m_attached || SUIsInvalid(other.m_camera)) {
52 return other.ref();
53 }
54 SUCameraRef new_camera = create_camera();
55 // Copy orientation
56 SUPoint3D position, target;
57 SUVector3D up_vector;
58 SUResult res = SUCameraGetOrientation(other.m_camera, &position, &target, &up_vector);
59 assert(res == SU_ERROR_NONE); _unused(res);
60 res = SUCameraSetOrientation(new_camera, &position, &target, &up_vector);
61 assert(res == SU_ERROR_NONE); _unused(res);
62 // Copy perspective/ortho setting
63 bool is_persp;
64 res = SUCameraGetPerspective(other.m_camera, &is_persp);
65 assert(res == SU_ERROR_NONE); _unused(res);
66 res = SUCameraSetPerspective(new_camera, is_persp);
67 assert(res == SU_ERROR_NONE); _unused(res);
68 if (is_persp) {
69 double fov_val;
70 res = SUCameraGetPerspectiveFrustumFOV(other.m_camera, &fov_val);
71 if (res == SU_ERROR_NONE) {
72 SUCameraSetPerspectiveFrustumFOV(new_camera, fov_val);
73 }
74 } else {
75 double height;
76 res = SUCameraGetOrthographicFrustumHeight(other.m_camera, &height);
77 if (res == SU_ERROR_NONE) {
78 SUCameraSetOrthographicFrustumHeight(new_camera, height);
79 }
80 }
81 return new_camera;
82}

163void Camera::orientation(Point3D& position, Point3D& target, Vector3D& up_vector) const {
164 if (!(*this)) {
165 throw std::logic_error("CW::Camera::orientation(): Camera is null");
166 }
167 SUPoint3D pos, tgt;
168 SUVector3D up;
169 SUResult res = SUCameraGetOrientation(m_camera, &pos, &tgt, &up);
170 assert(res == SU_ERROR_NONE); _unused(res);
171 position = Point3D(pos);
172 target = Point3D(tgt);
173 up_vector = Vector3D(up);

472Camera Camera::copy() const {
473 if (!(*this)) {
474 throw std::logic_error("CW::Camera::copy(): Camera is null");
475 }
476 Camera new_camera(create_camera(), false);
477 new_camera.set_orientation(this->eye(), this->target(), this->up());
478 new_camera.perspective(this->perspective());
479 if (this->perspective()) {
480 new_camera.fov(this->fov());
481 } else {
482 new_camera.orthographic_height(this->orthographic_height());
483 }
484 new_camera.fov_is_height(this->fov_is_height());
485 new_camera.image_width(this->image_width());
486 new_camera.description(this->description());
487 new_camera.is_2d(this->is_2d());
488 new_camera.scale_2d(this->scale_2d());
489 new_camera.center_2d(this->center_2d());
490 return new_camera;
491}
