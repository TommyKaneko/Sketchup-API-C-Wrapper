41/**
42 * @brief C++ wrapper for SUDimensionStyleRef.
43 *
44 * A DimensionStyle defines the default appearance of dimension entities
45 * in a model. There is one DimensionStyle per model, accessible via
46 * Model::dimension_style().
47 *
48 * This is a read-only class — all properties are getters only.
49 * DimensionStyle has no Create/Release functions; it is owned by the model.
50 *
51 * @see SUDimensionStyleRef
52 * @since SketchUp 2017, API 5.0
53 */
54class DimensionStyle {

72 * @brief Returns the SUDimensionStyleRef object.

151 bool highlight_non_associative() const;
