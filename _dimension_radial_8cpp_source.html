80void DimensionRadial::curve_instance_path(const InstancePath& path) {
81 if (!(*this)) {
82 throw std::logic_error("CW::DimensionRadial::curve_instance_path(): DimensionRadial is null");
83 }
84 SUResult res = SUDimensionRadialSetCurveInstancePath(this->radial_ref(), path.ref());
85 switch (res) {
86 case SU_ERROR_NONE:
87 break;
88 case SU_ERROR_INVALID_ARGUMENT:
89 throw std::invalid_argument("CW::DimensionRadial::curve_instance_path(): Invalid instance path");
90 case SU_ERROR_GENERIC:
91 throw std::invalid_argument("CW::DimensionRadial::curve_instance_path(): Instance path leaf is not an arc curve");
92 default:
93 assert(res == SU_ERROR_NONE); _unused(res);
94 }
95}
