52SUComponentDefinitionRef ComponentDefinition::copy_reference(const ComponentDefinition& other) {
53 if (other.m_attached && SUIsValid(other.m_entity)) {
54 return other.ref();
55 }
56 // The other definition has not been attached to the model, so copy its properties to a new object
57 SUComponentDefinitionRef new_definition = create_definition();
58 if (SUIsValid(other.m_entity)) {
59 // Copy across all nested geometry
60 SUEntitiesRef new_entities_ref = SU_INVALID;
61 SUResult res = SUComponentDefinitionGetEntities(new_definition, &new_entities_ref);
62 assert(res == SU_ERROR_NONE); _unused(res);
63 #if SketchUpAPI_VERSION_MAJOR < 2021
64 Entities new_entities(new_entities_ref, other.model().ref());
65 #else
66 Entities new_entities(new_entities_ref);
67 #endif
68 new_entities.add(other.entities());
69 }
70 return new_definition;
71}

107ComponentDefinition& ComponentDefinition::operator=(const ComponentDefinition& other) {
108 if (!m_attached && SUIsValid(m_entity)) {
109 SUComponentDefinitionRef definition = this->ref();
110 SUResult res = SUComponentDefinitionRelease(&definition);
111 assert(res == SU_ERROR_NONE); _unused(res);
112 }
113 m_entity = SUComponentDefinitionToEntity(copy_reference(other));
114 // Copy across all ComponentDefinition properties
115 this->name(other.name());
116 this->behavior(other.behavior());
117 DrawingElement::operator=(other);
118 return *this;
119}

246std::vector<ComponentInstance> ComponentDefinition::instances() const {
247 if (!SUIsValid(this->ref())) {
248 throw std::logic_error("CW::ComponentDefinition::instances(): definition is null");
249 }
250 size_t count = this->num_instances();
251 if (count == 0) {
252 return std::vector<ComponentInstance>{};
253 }
254 std::vector<SUComponentInstanceRef> instance_refs(count, SU_INVALID);
255 SUResult res = SUComponentDefinitionGetInstances(this->ref(), count, instance_refs.data(), &count);
256 assert(res == SU_ERROR_NONE); _unused(res);
257 std::vector<ComponentInstance> instances(count);
258 std::transform(instance_refs.begin(), instance_refs.end(), instances.begin(),
259 [](const SUComponentInstanceRef& value) {
260 return ComponentInstance(value);
261 });;
262 return instances;
263}

277std::vector<Opening> ComponentDefinition::openings() const {
278 if (!(*this)) {
279 throw std::logic_error("CW::ComponentDefinition::num_openings(): ComponentDefinition is null");
280 }
281 size_t count = this->num_openings();
282 std::vector<SUOpeningRef> opening_refs(count, SU_INVALID);
283 size_t return_count = 0;
284 SU_RESULT res = SUComponentDefinitionGetOpenings(this->ref(), count, opening_refs.data(), &return_count);
285 assert(return_count == count);
286 assert(res == SU_ERROR_NONE); _unused(res);
287 std::vector<Opening> openings;
288 openings.reserve(return_count);
289 for (SUOpeningRef p : opening_refs) {
290 openings.push_back(Opening(p));
291 }
292 return openings;
293}
