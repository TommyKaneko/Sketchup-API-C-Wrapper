59 /**
60 @brief Converts a Ruby API entity to a Cpp API entity.

62 @param ruby_entity The Ruby API entity reference.
63 @return The retrieved Cpp API entity reference.
64 */
65 static Entity entity_from_ruby(const RUBY_VALUE& ruby_entity);
66
67 /**
68 @brief Converts a Cpp API entity to a Ruby API imagerep.
69 @note Ruby takes ownership of the object it should not be released by
70 SUImageRepRelease. It will be released when Ruby garbage collects the
71 Ruby references to it.
72 @since SketchUp 2020.2, API 8.2
73 @param imagerep The Cpp API imagerep reference.
74 @return The retrieved Ruby API imagerep reference.
75 */
76 static RUBY_VALUE imagerep_to_ruby(const ImageRep& imagerep);
77
78 /**
79 @brief Converts a Ruby API imagerep to a Cpp API imagerep.
80 @note Ruby retains the ownership of the object it should not be released by
81 SUImageRepRelease.
82 @warning The returned C API reference points to an object that is owned by the
83 Ruby runtime and it may be garbage collected. To avoid this, ensure
84 that the Ruby API reference has a longer lifetime than the C API
85 reference.
86 @since SketchUp 2020.2, API 8.2
87 @param ruby_imagerep The Ruby API imagerep reference.
88 @return The retrieved C API imagerep reference.
89 */
90 static ImageRep imagerep_from_ruby(const RUBY_VALUE& ruby_entity);
