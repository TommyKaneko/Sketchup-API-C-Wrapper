33// Implicit copying - add source entities into an empty model's entities
34TEST_F(ModelLoad, GroupCopying)
35{
36 using namespace CW;
37
38 Entities src_entities = m_model->entities();
39 std::vector<Group> groups = src_entities.groups();
40 ASSERT_GT(groups.size(), (size_t)0);
41 // Add all entities (including groups) from source into the empty copy
42 Entities dest_entities = m_model_copy->entities();
43 for (CW::Group &group : groups) {
44 Group new_group = dest_entities.add_group(group.definition(), group.transformation());
45 new_group.name(group.name());
46 // TODO: apply materials and layers to new group
47 }
48 std::vector<Group> copied_groups = m_model_copy->entities().groups();
49
50 SaveModel("GroupImplicitCopying");
51 EXPECT_EQ(copied_groups.size(), groups.size());
52
53 for (size_t i = 0; i < groups.size(); ++i) {
54 // Must be different native objects
55 EXPECT_NE(groups[i].ref().ptr, copied_groups[i].ref().ptr);
56 ASSERT_NO_FATAL_FAILURE(GroupsAreEqual(groups[i], copied_groups[i]));
57 }
58}
