39/**
40 * @brief C++ wrapper for SUSchemaRef.
41 *
42 * A Schema represents a classification schema loaded into the model's
43 * Classifications manager. It provides access to schema types by name
44 * and the schema's own name.
45 *
46 * This object is not owned — it references data managed by the
47 * Classifications object and must not be released.
48 *
49 * @see SUSchemaRef
50 * @since SketchUp 2015, API 3.0
51 */
52class Schema {
53 private:
54 SUSchemaRef m_schema;
55
56 public:
57 /**
58 * @brief Constructs a NULL Schema object.
59 */
60 Schema();
61
62 /**
63 * @brief Constructs from an existing SUSchemaRef.
64 * @param schema - the SUSchemaRef obtained from Classifications.
65 */
66 Schema(SUSchemaRef schema);
67
68 /**
69 * @brief Returns the underlying SUSchemaRef.
70 */
71 SUSchemaRef ref() const;
72
73 /**
74 * @brief Returns true if this is an invalid/null schema.
75 */
76 bool operator!() const;
77
78 /**
79 * @brief Retrieves the name of this schema.
80 * @since SketchUp 2022.0, API 10.0
81 * @throws std::logic_error if the schema is null.
82 */
83 String name() const;
84
85 /**
86 * @brief Retrieves a schema type by name from this schema.
87 * @since SketchUp 2015, API 3.0
88 * @param schema_type_name - the name of the schema type.
89 * @return SUSchemaTypeRef for the requested type (may be invalid if not found).
90 * @throws std::logic_error if the schema is null.
91 */
92 SUSchemaTypeRef get_schema_type(const std::string& schema_type_name) const;
93};
