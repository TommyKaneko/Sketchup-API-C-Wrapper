148// Model-based tests: copy guide points from source model to destination model
149class GuidePointModelTest : public ModelLoad {};
150
151
152TEST_F(GuidePointModelTest, CopyGuidePoints) {
153 // Read guide points from the source model
154 CW::Entities src_entities = m_model->entities();
155 std::vector<CW::GuidePoint> source_points = src_entities.guide_points();
156 ASSERT_GT(source_points.size(), 0u);
157
158 // Copy each guide point using copy() and add to the destination model
159 std::vector<CW::GuidePoint> new_points;
160 for (const auto& sp : source_points) {
161 new_points.push_back(sp.copy());
162 }
163 CW::Entities dest_entities = m_model_copy->entities();
164 dest_entities.add_guide_points(new_points);
165
166 // Read back from destination and verify positions match
167 std::vector<CW::GuidePoint> copied_points = dest_entities.guide_points();
168 ASSERT_EQ(copied_points.size(), source_points.size());
169
170 for (size_t i = 0; i < source_points.size(); ++i) {
171 CW::Point3D src_pos = source_points[i].position();
172 CW::Point3D dest_pos = copied_points[i].position();
173 EXPECT_DOUBLE_EQ(src_pos.x, dest_pos.x);
174 EXPECT_DOUBLE_EQ(src_pos.y, dest_pos.y);
175 EXPECT_DOUBLE_EQ(src_pos.z, dest_pos.z);
176 }
177
178 SaveModel("GuidePoints");
179}
