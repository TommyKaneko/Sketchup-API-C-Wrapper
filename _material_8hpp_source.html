42/**
43 * @brief Wrapper for SUMaterialRef.
44 *
45 * Represents a material that can be applied to drawing elements. A material
46 * can have a color, texture, opacity, and type (colored, textured, or
47 * colorized texture).
48 *
49 * @see SUMaterialRef in the SketchUp C API
50 */
51class Material :public Entity {
52 private:
53 static SUMaterialRef create_material();
54
55 /**
56 * @brief Creates a SUMaterialRef derived from an existing Material.
57 *
58 * If the Material is already attached to a model, its existing ref is
59 * returned. Otherwise a new SUMaterialRef is created.
60 *
61 * @param mat The Material to derive from.
62 * @return The SUMaterialRef.
63 */
64 static SUMaterialRef copy_reference(const Material& mat);
65
66 public:
67 /**
68 * @brief Constructs a null Material.
69 */
70 Material();
71
72 /**
73 * @brief Constructs a new Material with the given name.
74 * @param name The name for the material (UTF-8 encoded).
75 */
76 Material(String name);
77
78 /**
79 * @brief Constructs a Material from an existing SUMaterialRef.
80 * @param material_ref The native material reference.
81 * @param attached Whether the material is already owned by a model.
82 */
83 Material(SUMaterialRef material_ref, bool attached = true);
84
85 /**
86 * @brief Copy constructor.
87 * @param other The Material to copy.
88 */
89 Material(const Material& other);
90
91 /**
92 * @brief Copy assignment operator.
93 * @param other The Material to assign from.
94 * @return Reference to this object.
95 */
96 Material& operator=(const Material& other);
97
98 /**
99 * @brief Destructor. Releases the material if it is not attached to a model.
100 */
101 ~Material();
102
103 /**
104 * @brief Returns the wrapped SUMaterialRef.
105 * @return The raw SUMaterialRef.
106 */
107 SUMaterialRef ref() const;
108
109 /** @brief Implicit conversion to SUMaterialRef. */
110 operator SUMaterialRef() const;
111
112 /** @brief Implicit conversion to SUMaterialRef*. */
113 operator SUMaterialRef*();
114
115 /**
116 * @brief Returns true if this is an invalid or unset material.
117 * @return True if the material is null or has no entity reference.
118 */
119 bool operator !() const;
120
121 /**
122 * @brief Returns a copy of this Material, not attached to any model.
123 * @return A new unattached Material with the same properties.

125 Material copy() const;
126
127 /**
128 * @brief Retrieves the color of the material.
129 * @return The Color.
130 * @throws std::logic_error If the material is null.
131 * @see SUMaterialGetColor

133 Color color() const;
134
135 /**
136 * @brief Sets the color of the material.
137 * @param color The Color to set.
138 * @throws std::logic_error If the material is null.

140 */
141 void color(const Color& color);
142
143 /**
144 * @brief Retrieves the internal name of the material.
145 *
146 * The internal name is the unprocessed identifier string stored with
147 * the material.
148 *
149 * @return The name as a String.
150 * @throws std::logic_error If the material is null.
151 * @see SUMaterialGetName
152 */
153 String name() const;
154
155 /**
156 * @brief Sets the name of the material.
157 * @param string The name to set (UTF-8 encoded).
158 * @throws std::logic_error If the material is null.
159 * @see SUMaterialSetName
160 */
161 void name(const String& string);
162
163 /**
164 * @brief Retrieves the display name of the material.
165 *
166 * If the internal name is encased in square brackets, this returns the
167 * name without brackets. Otherwise it matches the name from GetName.
168 *
169 * @return The display name as a String.
170 * @throws std::logic_error If the material is null.
171 * @since SketchUp 2017, API 5.0
172 * @see SUMaterialGetNameLegacyBehavior
173 */
174 String display_name() const;
175
176 /**
177 * @brief Retrieves the opacity of the material.
178 * @return The alpha value in the range [0.0, 1.0].

180 * @see SUMaterialGetOpacity
181 */
182 double opacity() const;
183
184 /**
185 * @brief Sets the opacity of the material.
186 * @param alpha The alpha value in the range [0.0, 1.0].
187 * @throws std::logic_error If the material is null.
188 * @see SUMaterialSetOpacity
189 */
190 void opacity(const double alpha);
191

198 Texture texture() const;
199
200 /**
201 * @brief Sets the texture of the material.
202 * @param texture The Texture to set.
203 * @throws std::logic_error If the material is null.
204 * @see SUMaterialSetTexture
205 */
206 void texture(Texture& texture);
207
208 /**
209 * @brief Retrieves the type of the material.
210 * @return The SUMaterialType (e.g. colored, textured, colorized texture).
211 * @throws std::logic_error If the material is null.
212 * @see SUMaterialGetType
213 */
214 SUMaterialType type() const;
215
216 /**
217 * @brief Sets the type of the material.
218 * @param material_type The SUMaterialType to set.
219 * @throws std::logic_error If the material is null.
220 * @see SUMaterialSetType
221 */
222 void type(const SUMaterialType& material_type);
223
224 /**
225 * @brief Returns whether the opacity value is used.
226 * @return True if alpha is used.
227 * @throws std::logic_error If the material is null.
228 * @see SUMaterialGetUseOpacity
229 */
230 bool use_alpha() const;
231
232 /**
233 * @brief Sets whether the opacity value is used.
234 * @param flag True to use alpha, false otherwise.
235 * @throws std::logic_error If the material is null.
236 * @see SUMaterialSetUseOpacity
237 */
238 void use_alpha(bool flag);
239
