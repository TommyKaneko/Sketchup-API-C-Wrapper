81std::vector<Face> Entities::faces() const {
82 if (!SUIsValid(m_entities)) {
83 throw std::logic_error("CW::Entities::faces(): Entities is null");
84 }
85 size_t count = 0;
86 SUResult res = SUEntitiesGetNumFaces(m_entities, &count);
87 assert(res == SU_ERROR_NONE);
88 if (count == 0) {
89 return std::vector<Face>(0);
90 }
91 std::vector<SUFaceRef> face_refs(count, SU_INVALID);
92 res = SUEntitiesGetFaces(m_entities, count, face_refs.data(), &count);
93 assert(res == SU_ERROR_NONE); _unused(res);
94 std::vector<Face> faces(count);
95 std::transform(face_refs.begin(), face_refs.end(), faces.begin(),
96 [](const SUFaceRef& value) {
97 return Face(value);
98 });
99 return faces;
100}

205void Entities::add(const Entities& other) {
206 if (!SUIsValid(m_entities) || !SUIsValid(other.m_entities)) {
207 throw std::logic_error("CW::Entities::add(): Entities is null");
208 }
209 Model model = this->model();
210 GeometryInputPlus geom_input(&model);
211 geom_input.add_faces(other.faces());
212 geom_input.add_edges(other.edges());
213 this->fill(geom_input);
214 std::vector<ComponentInstance> instances = other.instances();
215 for (size_t i=0; i < instances.size(); ++i) {
216 this->add_instance(instances[i].definition(), instances[i].transformation(), instances[i].name());
217 }
218 std::vector<Group> other_groups = other.groups();
219 for (size_t i=0; i < other_groups.size(); ++i) {
220 Group new_group = this->add_group();
221 new_group.entities().add(other_groups[i].entities());
222 new_group.transformation(other_groups[i].transformation());
223 new_group.name(other_groups[i].name());
224 }
225 // TODO: other geometry types need to be added.
226}

242std::vector<Face> Entities::add_faces(std::vector<Face>& faces) {
243 if (!SUIsValid(m_entities)) {
244 throw std::logic_error("CW::Entities::add_faces(): Entities is null");
245 }
246 for (const auto& face : faces) {
247 if (face.attached()) {
248 throw std::invalid_argument("CW::Entities::add_faces(): Face is already attached to a model. Use Face::copy() to create a detached copy first.");
249 }
250 }
251 std::vector<SUFaceRef> refs(faces.size(), SU_INVALID);
252 std::transform(faces.begin(), faces.end(), refs.begin(), [](const CW::Face& face) {return face.ref(); });
253
254 SUResult res = SUEntitiesAddFaces(m_entities, refs.size(), refs.data());
255 assert(res == SU_ERROR_NONE); _unused(res);
256
257 // Transfer ownership of each face
258 for (auto& face : faces)
259 face.attached(true);
260
261 return faces;
262}

265std::vector<Edge> Entities::add_edges(std::vector<Edge>& edges) {
266 if (!SUIsValid(m_entities)) {
267 throw std::logic_error("CW::Entities::add_edges(): Entities is null");
268 }
269 std::vector<Edge> existing_edges = this->edges();
270 std::vector<Edge> edges_to_add; edges_to_add.reserve(edges.size());
271 for (Edge& edge : edges) {
272 if (!edge) {
273 continue;
274 }
275 // Check that each edge is not attached to another entities object
276 if (edge.attached()) {
277 edges_to_add.push_back(edge.copy());
278 }
279 else {
280 edges_to_add.push_back(edge);
281 }
282 }
283
284 std::vector<SUEdgeRef> refs(edges_to_add.size(), SU_INVALID);
285 std::transform(edges_to_add.begin(), edges_to_add.end(), refs.begin(), [](const CW::Edge& edge) {return edge.ref(); });
286
287 SUResult res = SUEntitiesAddEdges(m_entities, refs.size(), refs.data());
288 assert(res == SU_ERROR_NONE); _unused(res);
289 // Transfer ownership of each edge
290 for (auto& edge : edges_to_add)
291 edge.attached(true);
292 return edges_to_add;
293}

327ComponentInstance Entities::add_instance(const ComponentDefinition& definition, const Transformation& transformation, const String& name){
328 if (!SUIsValid(m_entities)) {
329 throw std::logic_error("CW::Entities::add_instance(): Entities is null");
330 }
331 if (!definition) {
332 throw std::invalid_argument("CW::Entities::add_instance(): ComponentDefinition argument is invalid");
333 }
334 if (!definition.attached()) {
335 throw std::invalid_argument("CW::Entities::add_instance(): ComponentDefinition must be attached to a model before creating an instance.");
336 }
337 // Check that the definition is attached to this model
338 if (definition.model().ref().ptr != this->model().ref().ptr) {
339 throw std::invalid_argument("CW::Entities::add_instance(): ComponentDefinition is attached to a different model than this Entities object.");
340 }
341 SUComponentInstanceRef instance = SU_INVALID;
342 SUResult res = SUComponentDefinitionCreateInstance(definition.ref(), &instance);
343 assert(res == SU_ERROR_NONE);
344 SUTransformation transform = transformation.ref();
345 res = SUComponentInstanceSetTransform(instance, &transform);
346 assert(res == SU_ERROR_NONE);
347 if (name.empty()) {
348 res = SUEntitiesAddInstance(m_entities, instance, NULL);
349 }
350 else {
351 SUStringRef name_ref = name.ref();
352 res = SUEntitiesAddInstance(m_entities, instance, &name_ref);
353 }
354 assert(res == SU_ERROR_NONE); _unused(res);
355 return ComponentInstance(instance, true);
356}

361Group Entities::add_group(const ComponentDefinition& definition, const Transformation& transformation) {
362 if (!SUIsValid(m_entities)) {
363 throw std::logic_error("CW::Entities::add_group(): Entities is null");
364 }
365 if (!definition) {
366 throw std::invalid_argument("CW::Entities::add_group(): ComponentDefinition argument is invalid");
367 }
368 if (!definition.is_group()) {
369 throw std::invalid_argument("CW::Entities::add_group(): ComponentDefinition given is not a group");
370 }
371 // Groups cannot be created with a component definition and transformation objects. Instead, the geometry must be copied in to a new Entities object in the group.
372 Group new_group = this->add_group();
373 new_group.transformation(transformation);
374 Entities group_entities = new_group.entities();
375 Entities def_entities = definition.entities();
376 // Add geometry one by one to Geometry input.
377 Model model = this->model();
378 GeometryInputPlus geom_input(&model);
379 // GeometryInputPlus object must first load layers and materials from the definition's model.
380 // TODO: this is unsatisfatory for performance - need a better way to handle this.
381 geom_input.load_materials(definition.model().materials());
382 geom_input.load_layers(definition.model().layers());
383 std::vector<Face> def_faces = def_entities.faces();
384 for (size_t i=0; i < def_faces.size(); ++i) {
385 //geom_input.add_face(def_faces[i]);
386 geom_input.add_face(def_faces[i], false);
387 }
388 std::vector<Edge> def_edges = def_entities.edges(true); // add stray edges only
389 for (size_t i=0; i < def_edges.size(); ++i) {
390 geom_input.add_edge(def_edges[i], def_edges[i].material(), def_edges[i].layer());
391 }
392 if((def_faces.size() + def_edges.size()) > 0) {
393 group_entities.fill(geom_input);
394 }
395 // Also add instances and groups
396 std::vector<Group> def_groups = def_entities.groups();
397 for (size_t i=0; i < def_groups.size(); ++i) {
398 group_entities.add_group(def_groups[i].definition() , def_groups[i].transformation());
399 }
400 std::vector<ComponentInstance> def_instances = def_entities.instances();
401 for (size_t i=0; i < def_instances.size(); ++i) {
402 group_entities.add_instance(def_instances[i].definition(), def_instances[i].transformation());
403 }
404
405 // TODO: add other entities to the group (construction lines, curves, etc)
406 return new_group;
407
408 // TODO: the way groups are implemented are a problem. Come back to this.
409 //SUResult res = SUComponentDefinitionCreateGroup(definition.ref(), &group);
410 //assert(res == SU_ERROR_NONE); _unused(res);
411 //SUTransformation transform = transformation.ref();
412 //res = SUComponentInstanceSetTransform(instance, &transform);
413}
