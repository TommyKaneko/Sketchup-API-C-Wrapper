40/**
41 * @brief C++ wrapper for SUFontRef.
42 *
43 * A Font object represents a font used by text and dimension entities.
44 * Font objects are read-only — they are obtained from Text or Dimension
45 * entities and cannot be created or released directly.
46 *
47 * @see SUFontRef
48 * @since SketchUp 2018, API 6.0
49 */
50class Font :public Entity {
51 public:
52 /**
53 * @brief Constructs a NULL Font object.
54 */
55 Font();
56
57 /**
58 * @brief Constructs a Font from an existing SUFontRef.
59 * @since SketchUp 2018, API 6.0
60 * @param font - SUFontRef object.
61 * @param attached - flag indicating whether the font is owned by
62 * a parent object (e.g. a Text or Dimension entity).
63 */
64 Font(SUFontRef font, bool attached = true);
65
66 /**
67 * @brief Returns the SUFontRef object.
68 */
69 SUFontRef ref() const;
70 operator SUFontRef() const;
71 operator SUFontRef*();
72
73 /**
74 * @brief Returns true if this is an invalid/null font.
75 */
76 bool operator!() const;
77
78 /**
79 * @brief Retrieves the font face name (e.g. "Arial", "Helvetica").
80 * @since SketchUp 2018, API 6.0
81 * @throws std::logic_error if the font is null.
82 */
83 String face_name() const;
84
85 /**
86 * @brief Retrieves the font point size.
87 * @since SketchUp 2018, API 6.0
88 * @throws std::logic_error if the font is null.
89 */
90 size_t point_size() const;
91
92 /**
93 * @brief Retrieves whether the font is bold.
94 * @since SketchUp 2018, API 6.0
95 * @throws std::logic_error if the font is null.
96 */
97 bool bold() const;
98
99 /**
100 * @brief Retrieves whether the font is italic.
101 * @since SketchUp 2018, API 6.0
102 * @throws std::logic_error if the font is null.
103 */
104 bool italic() const;
105
106 /**
107 * @brief Retrieves whether the font uses world size (true) or screen size
108 * (false).
109 * @since SketchUp 2019, API 7.0
110 * @throws std::logic_error if the font is null.
111 */
112 bool use_world_size() const;
113
114 /**
115 * @brief Retrieves the world size of the font in inches when use_world_size()
116 * is true.
117 * @since SketchUp 2019, API 7.0
118 * @throws std::logic_error if the font is null.
119 */
120 double world_size() const;
121};
