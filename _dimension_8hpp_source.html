45/**
46 * @brief C++ wrapper for SUDimensionRef.
47 *
48 * A Dimension is the base class for linear and radial dimension entities.
49 * Dimensions are drawing elements that display measurement annotations in
50 * the model. This class provides access to the properties common to all
51 * dimension types (text, plane, 3D flag, arrow type, font).
52 *
53 * @note The SketchUp C API does not provide SUEntitiesAddDimension(). Dimensions
54 * can be created as DimensionLinear or DimensionRadial and retrieved from
55 * entities, but there is no general add-dimension method.
56 *
57 * @see SUDimensionRef
58 * @since SketchUp 2017, API 5.0
59 */
60class Dimension :public DrawingElement {
61 public:
62 /**
63 * @brief Constructs a NULL Dimension object.
64 */
65 Dimension();
66
67 /**
68 * @brief Constructs a Dimension from an existing SUDimensionRef.
69 * @since SketchUp 2017, API 5.0
70 * @param dimension - SUDimensionRef object.
71 * @param attached - flag indicating whether the dimension is owned by
72 * a parent object (e.g. SUEntitiesRef).
73 */
74 Dimension(SUDimensionRef dimension, bool attached = true);
75
76 /** @brief Copy constructor */
77 Dimension(const Dimension& other);
78
79 /** @brief Copy assignment operator */
80 Dimension& operator=(const Dimension& other);

82 /**
83 * @brief Returns the SUDimensionRef object.
84 */
85 SUDimensionRef ref() const;
86 operator SUDimensionRef() const;
87 operator SUDimensionRef*();
88
89 /**
90 * @brief Returns true if this is an invalid/null dimension.
91 */
92 bool operator!() const;
93
94 /**
95 * @brief Returns the dimension as a DimensionLinear object. This is effectively an upcast operation. If the dimension is not linear, an exception is thrown.
96 * @return DimensionLinear object with the same underlying SUDimensionRef as this dimension.
97 * @throws std::logic_error if the dimension is null or not linear.
98 */
99 DimensionLinear as_linear() const;

101 /**

103 * @return DimensionRadial object with the same underlying SUDimensionRef as this dimension.
104 * @throws std::logic_error if the dimension is null or not radial.
105 */
106 DimensionRadial as_radial() const;
107
108 /**

113 SUDimensionType type() const;

115 /**

121
122 /**
123 * @brief Sets the text of the dimension. Assumed to be UTF-8 encoded.

125 * @param text - the text content to set.
126 * @throws std::logic_error if the dimension is null.
127 */
128 void text(const String& text);
129 void text(const std::string& text);
130
131 /**
132 * @brief Retrieves the plane of the dimension.
133 * @since SketchUp 2017, API 5.0
134 * @throws std::logic_error if the dimension is null.
135 */
136 Plane3D plane() const;
137
138 /**
139 * @brief Retrieves whether the dimension text is 3D (positioned in model
140 * space rather than screen space).
141 * @since SketchUp 2017, API 5.0
142 * @throws std::logic_error if the dimension is null.
143 */
144 bool text_3d() const;
145
146 /**
147 * @brief Sets whether the dimension text is 3D.
148 * @since SketchUp 2017, API 5.0
149 * @param is_3d - true for 3D text, false for screen-space text.
150 * @throws std::logic_error if the dimension is null.
151 */
152 void text_3d(bool is_3d);
153
154 /**
155 * @brief Retrieves the dimension's arrow type.
156 * @since SketchUp 2017, API 5.0
157 * @throws std::logic_error if the dimension is null.
158 */
159 SUArrowType arrow_type() const;
160
161 /**
162 * @brief Sets the dimension's arrow type.
163 * @since SketchUp 2017, API 5.0
164 * @param type - the arrow type to set.
165 * @throws std::logic_error if the dimension is null.
166 * @throws std::out_of_range if the arrow type is not valid.
167 */
168 void arrow_type(SUArrowType type);
169
170 /**
171 * @brief Retrieves the dimension's font.
172 * @since SketchUp 2019, API 7.0
173 * @throws std::logic_error if the dimension is null.
174 */
175 Font font() const;
176
177 /**
178 * @brief Sets the dimension's font.
179 * @since SketchUp 2019, API 7.0
180 * @param font - the Font to set.
181 * @throws std::logic_error if the dimension is null.
182 */
183 void font(const Font& font);
184};
