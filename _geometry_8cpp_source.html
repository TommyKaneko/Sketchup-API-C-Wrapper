50Radians::Radians(const double &rhs):
51 m_val(rhs)
52{
53 if(rhs > PI2) {
54 m_val = std::fmod(rhs, PI2);
55 }
56 else if(rhs < 0.0) {
57 m_val = PI2 - std::fmod(std::fabs(rhs), PI2);
58 }
59}

289double Vector3D::angle(const Vector3D& vector_b) const {
290 assert(!is_null);
291 // Check that acos doesn't suffer domain error as a result of being slightly outside the range of -1 to +1
292 double dot_product = unit().dot(vector_b.unit());
293 if (dot_product < -1.0) {
294 dot_product = -1.0;
295 }
296 else if (dot_product > 1.0) {
297 dot_product = 1.0;
298 }
299 return acos(dot_product);
300}

320Vector3D::Colinearity Vector3D::colinear(const Vector3D& vector_b) const {
321 if (this->length() < EPSILON || vector_b.length() < EPSILON) {
322 return Vector3D::Colinearity::UNDEFINED;
323 }
324 Vector3D combined = (*this) + vector_b;
325 double combined_length = combined.length();
326 double added_length = (*this).length() + vector_b.length();
327 if (std::fabs(added_length - combined_length) < EPSILON) {
328 return Vector3D::Colinearity::COLINEAR_PRO;
329 }
330 double subtracted_length = (*this).length() - vector_b.length();
331 if (std::fabs(subtracted_length - combined_length) < EPSILON) {
332 return Vector3D::Colinearity::COLINEAR_ANTI;
333 }
334 return Vector3D::Colinearity::NO;
335}

338Vector3D Vector3D::rotate_about(double angle, const Vector3D& axis) const {
339 // This solution is derived from this page: http://math.stackexchange.com/questions/511370/how-to-rotate-one-vector-about-another
340 double b_dot_b = axis.dot(axis);
341 Vector3D a_component_b_dir = ((*this).dot(axis) * b_dot_b) * axis;
342 Vector3D a_component_b_orth = (*this) - (((*this).dot(axis) / b_dot_b) * axis);
343 Vector3D w = axis.cross(a_component_b_orth);
344 double x1 = cos(angle) / a_component_b_orth.length();
345 double x2 = sin(angle) / w.length();
346 Vector3D a_component_b_orth_rot = a_component_b_orth.length() * ((x1 * a_component_b_orth) + (x2 * w));
347 return a_component_b_orth_rot + a_component_b_dir;
348}

497Point3D Point3D::intersection_between_lines(const Point3D& point_a, const Vector3D& vector_a, const Point3D& point_b, const Vector3D& vector_b, bool return_colinear) {
498 if (vector_a.length() < EPSILON || vector_b.length() < EPSILON) {
499 throw std::invalid_argument("CW::Point3D::intersection_between_lines() - cannot find intersection of lines with zero length");
500 }
501 // Find the closest points according to this solution: http://paulbourke.net/geometry/pointlineplane/
502 // And this: http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
503 Vector3D a_to_b = point_b - point_a;
504 Vector3D vec_a_cross_b = vector_a.unit().cross(vector_b.unit());
505 Vector3D a_to_b_cross_vec_a = a_to_b.unit().cross(vector_a.unit());
506 Vector3D zero_vector(0.0,0.0,0.0);
507 // Check for collinearity
508 if (vec_a_cross_b == zero_vector) {
509 if (a_to_b_cross_vec_a == zero_vector) {
510 // Lines are collinear, so there is no intersection
511 if (!return_colinear) {
512 return Point3D(false);
513 }
514 bool opposite_direction;
515 if (vector_a.unit() == vector_b.unit())
516 opposite_direction = false;
517 else
518 opposite_direction = true;
519 double vec_a_factor_0 = a_to_b.dot(vector_a) / vector_a.dot(vector_a);
520 double vec_a_factor_1 = vec_a_factor_0 + (vector_b.dot(vector_a) / vector_a.dot(vector_a));
521 double vec_a_epsilon = Vector3D::EPSILON / vector_a.length(); // Note accuracy needs to be determined
522 if ((!opposite_direction && vec_a_factor_0 < vec_a_epsilon && vec_a_factor_1 > -vec_a_epsilon) ||
523 (opposite_direction && vec_a_factor_1 < vec_a_epsilon && vec_a_factor_0 > -vec_a_epsilon)) {
524 // The intersection is at the start of line A
525 return point_a;
526 }
527 if ((!opposite_direction && vec_a_factor_0 > -vec_a_epsilon && vec_a_factor_0 < (1 + vec_a_epsilon))) {
528 // The intersection is at the start of line B
529 return point_b;
530 }
531 if (opposite_direction && vec_a_factor_1 > -vec_a_epsilon && vec_a_factor_1 < (1 + vec_a_epsilon)) {
532 // The intersection is at the end of line B
533 return point_b + vector_b;
534 }
535 // Lines are disjoint
536 return Point3D(false);
537
538 }
539 else {
540 // Lines are parallel, and so there is no intersection.
541 return Point3D(false);
542 }
543 }
544 // We create lines and find the closest points between them.
545 Line3D line_a(point_a, vector_a);
546 Line3D line_b(point_b, vector_b);
547 std::pair<Point3D, Point3D> closest_points = line_a.closest_points(line_b);
548 // Check closest points are actually the same.
549 if (closest_points.first != closest_points.second) {
550 // No intersection
551 return Point3D(false);
552 }
553 // Otherwise, we need to see if the intersection of the lines lie between the line segments
554 Point3D intersection = closest_points.first;
555 Vector3D a_to_int = intersection - point_a;
556 Vector3D b_to_int = intersection - point_b;
557 // Express the point of intersection as a scalar of the vectors from point A and point B
558 double a_direction_factor = 0.0;
559 if (a_to_int != zero_vector)
560 a_direction_factor = vector_a.unit().dot(a_to_int.unit());
561 double b_direction_factor = 0.0;
562 if (b_to_int != zero_vector)
563 b_direction_factor = vector_b.unit().dot(b_to_int.unit());
564
565 double a_factor = a_direction_factor * a_to_int.length() / vector_a.length();
566 double b_factor = b_direction_factor * b_to_int.length() / vector_b.length();
567 double a_epsilon = Vector3D::EPSILON / vector_a.length();
568 double b_epsilon = Vector3D::EPSILON / vector_b.length();
569 if (a_factor > -a_epsilon && a_factor < 1.0 + a_epsilon &&
570 b_factor > -b_epsilon && b_factor < 1.0 + b_epsilon) {
571 return intersection;
572 }
573 return Point3D(false);
574}

577Point3D Point3D::ray_line_intersection(const Point3D& point_a, const Vector3D& vector_a, const Point3D& point_b, const Vector3D& ray_b, bool return_colinear) {
578 // Solution to finding intersection between two line segments derived from this answer: http://stackoverflow.com/questions/563198/how-do-you-detect-where-two-line-segments-intersect
579 Vector3D zero_vector(0.0,0.0,0.0);
580 if (ray_b == zero_vector || vector_a == zero_vector) {
581 throw std::invalid_argument("CW::Point3D::ray_line_intersection() - given ray/line have zero length");
582 }
583 Vector3D a_to_b = point_b - point_a;
584 // Check if the start of the lines touch
585 if (a_to_b == zero_vector) {
586 return point_a;
587 }
588 Vector3D b_to_a = -a_to_b;
589 Vector3D vec_a_cross_b_z_comp = vector_a.unit().cross(ray_b.unit()); // Use unit vectors for easier comparisons with zero vector
590 Vector3D a_to_b_cross_vec_a_z_comp = a_to_b.unit().cross(vector_a.unit()); // Use unit vectors for easier comparisons with zero vector
591 // Check for collinearity
592 if (vec_a_cross_b_z_comp == zero_vector) {
593 if (a_to_b_cross_vec_a_z_comp == zero_vector) {
594 // Lines are collinear, so there is no intersection
595 if (!return_colinear) {
596 return Point3D(false);
597 }
598 double b_to_a_factor = b_to_a.dot(ray_b) / ray_b.dot(ray_b);
599 double b_to_a_end_factor = b_to_a_factor + (vector_a.dot(ray_b) / ray_b.dot(ray_b));
600 if ((b_to_a_factor < 0.0 && b_to_a_end_factor > 0.0) || (b_to_a_factor > 0.0 && b_to_a_end_factor < 0.0)) {
601 // The start of ray B is on line A
602 return point_b;
603 }
604 if (b_to_a_factor > 0.0) {
605 if (b_to_a_factor < b_to_a_end_factor) {
606 // The intersection is at the start of line A
607 return point_a;
608 }
609 else {
610 // The intersection is at the end of line B
611 return point_a + vector_a;
612 }
613 }
614 // Lines are disjoint
615 return Point3D(false);
616
617 }
618 else {
619 // Lines are parallel, and so there is no intersection.
620 return Point3D(false);
621 }
622 }
623 // We create lines and find the closest points between them.
624 Line3D line_a(point_a, vector_a);
625 Line3D line_b(point_b, ray_b);
626 std::pair<Point3D, Point3D> closest_points = line_a.closest_points(line_b);
627 // Check closest points are actually the same.
628 if (closest_points.first != closest_points.second) {
629 // No intersection
630 return Point3D(false);
631 }
632 // Otherwise, we need to see if the intersection of the lines lie between the line segments
633 Point3D intersection = closest_points.first;
634 Vector3D a_to_int = intersection - point_a;
635 Vector3D b_to_int = intersection - point_b;
636 // Express the point of intersection as a scalar of the vectors from point A and point B
637 double a_direction_factor = 0.0;
638 if (a_to_int != zero_vector)
639 a_direction_factor = vector_a.unit().dot(a_to_int.unit());
640 double b_direction_factor = 0.0;
641 if (b_to_int != zero_vector)
642 b_direction_factor = ray_b.unit().dot(b_to_int.unit());
643
644 double a_factor = a_direction_factor * a_to_int.length() / vector_a.length();
645 double b_factor = b_direction_factor * b_to_int.length() / ray_b.length();
646 double a_epsilon = Vector3D::EPSILON / vector_a.length();
647 double b_epsilon = Vector3D::EPSILON / ray_b.length();
648 if (a_factor > -a_epsilon && a_factor < 1.0 + a_epsilon &&
649 b_factor > -b_epsilon) {
650 return intersection;
651 }
652 return Point3D(false);
653
654 /*

785 const Vector3D line_vector = (*this).normal().cross(plane2.normal());

790 return Line3D(false);

793 ((*this).normal().cross(line_vector) * plane2.d)) / determinant;
794 return Line3D(line_point, line_vector);

1086 throw std::invalid_argument("CW::Line3D::intersection(): given plane is null");
1087 }
1088 if (!(*this)) {
1089 throw std::logic_error("CW::Line3D::intersection(): this line is null");
1090 }
1091 // @see http://paulbourke.net/geometry/pointlineplane/
1092 double numerator = (plane.a * this->point.x) + (plane.b * this->point.y) + (plane.c * this->point.z) + plane.d;
1093 double denominator = -(plane.a * this->direction.x) - (plane.b * this->direction.y) - (plane.c * this->direction.z);
1094 if (std::abs(denominator) < EPSILON) {
1095 // The line is parallel to or on the plane

1104std::pair<Point3D, Point3D> Line3D::closest_points(const Line3D &other_line) const {
1105 if (!other_line) {
1106 throw std::invalid_argument("CW::Line3D::closest_points(): given line is null");
1107 }
1108 if (!(*this)) {
1109 throw std::logic_error("CW::Line3D::closest_points(): this line is null");
1110 }
1111
1112 // @see http://paulbourke.net/geometry/pointlineplane/
1113 double d1343,d4321,d1321,d4343,d2121;
1114 double numer,denom;
1115 // p13 is a vector between two points on different lines
1116 Vector3D p13 = this->point - other_line.point;
1117 Vector3D p43 = other_line.direction;
1118 if (std::abs(p43.x) < EPSILON && std::abs(p43.y) < EPSILON && std::abs(p43.z) < EPSILON) {
1119 // The two lines are along each other
1120 return std::pair<Point3D,Point3D> (Point3D(false), Point3D(false));
1121 }
1122 Vector3D p21 = this->direction;
1123 //if (std::abs(p21.x) < EPSILON/2 && std::abs(p21.y) < EPSILON/2 && std::abs(p21.z) < EPSILON/2) {
1124 // The vector is zero - should not happen
1125 // return std::pair<Point3D,Point3D> (Point3D(false), Point3D(false));
1126 //}
1127 d1343 = p13.x * p43.x + p13.y * p43.y + p13.z * p43.z;
1128 d4321 = p43.x * p21.x + p43.y * p21.y + p43.z * p21.z;
1129 d1321 = p13.x * p21.x + p13.y * p21.y + p13.z * p21.z;
1130 d4343 = p43.x * p43.x + p43.y * p43.y + p43.z * p43.z;
1131 d2121 = p21.x * p21.x + p21.y * p21.y + p21.z * p21.z;
1132
1133 denom = d2121 * d4343 - d4321 * d4321;
1134 if (std::abs(denom) < EPSILON) {
1135 // Lines are parallel
1136 return std::pair<Point3D,Point3D> (Point3D(false), Point3D(false));
1137 }
1138 numer = d1343 * d4321 - d1321 * d4343;
1139
1140 double mua = numer / denom;
1141 double mub = (d1343 + d4321 * mua) / d4343;
1142 Point3D pa = this->point + (this->direction * mua);
1143 Point3D pb = other_line.point + (other_line.direction * mub);
1144 return std::pair<Point3D,Point3D> (pa, pb);
1145}

1169bool Line3D::on_line(const Point3D& test_point) const {
1170 if (!test_point) {
1171 throw std::invalid_argument("CW::Line3D::on_line(): given point is null");
1172 }
1173 if (!(*this)) {
1174 throw std::logic_error("CW::Line3D::on_line(): this line is null");
1175 }
1176 double factor;
1177 enum class OnLineFactorCoords {
1178 X,
1179 Y,
1180 Z
1181 };
1182 OnLineFactorCoords factor_val; // This is to save extra calculations.
1183 // To get the most accurate factor, use one of the larger values among x, y OR z to calculate the factor. The smallest possible value for all three x, y, z to have (given that direction is a unit vector) is 1 / sqrt(3) = 0.577
1184 if (this->direction.x > 0.56) {
1185 // Get the factor with which to multiply the x value, and see if it is the same for y and z values too.
1186 factor = (test_point.x - this->point.x) / this->direction.x;
1187 factor_val = OnLineFactorCoords::X;
1188 }
1189 else if (this->direction.y > 0.56) {
1190 factor = (test_point.y - this->point.y) / this->direction.y;
1191 factor_val = OnLineFactorCoords::Y;
1192 }
1193 else {
1194 factor = (test_point.z - this->point.z) / this->direction.z;
1195 factor_val = OnLineFactorCoords::Z;
1196 }
1197 if (factor_val != OnLineFactorCoords::X) {
1198 double x_test = this->point.x + (this->direction.x * factor);
1199 if (std::abs(x_test - test_point.x) > EPSILON) {
1200 return false;
1201 }
1202 }
1203 if (factor_val != OnLineFactorCoords::Y) {
1204 double y_test = this->point.y + (this->direction.y * factor);
1205 if (std::abs(y_test - test_point.y) > EPSILON) {
1206 return false;
1207 }
1208 }
1209 if (factor_val != OnLineFactorCoords::Z) {
1210 double z_test = this->point.z + (this->direction.z * factor);
1211 if (std::abs(z_test - test_point.z) > EPSILON) {
1212 return false;
1213 }
1214 }
1215 return true;
1216}

1219bool Line3D::on_line_segment(const Point3D& point_a, const Point3D& point_b, const Point3D& test_point) {
1220 if (!point_a || !test_point || !point_b) {
1221 throw std::invalid_argument("CW::Line3D::on_line_segment(): given point is null");
1222 }
1223 if (point_a == point_b) {
1224 throw std::invalid_argument("CW::Line3D::on_line_segment(): given points are equal");
1225 }
1226 // From: https://stackoverflow.com/questions/328107/how-can-you-determine-a-point-is-between-two-other-points-on-a-line-segment
1227 Vector3D a_to_b = point_b - point_a;
1228 Vector3D a_to_c = test_point - point_a;
1229 if (a_to_c == Vector3D::zero_vector()) {
1230 return true;
1231 }
1232 Vector3D crossproduct = a_to_b.unit().cross(a_to_c.unit());
1233 if (fabs(crossproduct.length()) > EPSILON) {
1234 // Not aligned
1235 return false;
1236 }
1237 // Check that the distance matches.
1238 double dotproduct = a_to_b.dot(a_to_c);
1239 if (dotproduct < EPSILON) {
1240 return false;
1241 }
1242
1243 double squaredlengthba = pow(a_to_b.length()+EPSILON, 2);
1244 if (dotproduct > squaredlengthba) {
1245 return false;
1246 }
1247 return true;
1248}

1289
1290bool operator==(const Line3D& lhs, const Line3D& rhs) {
1291 if (lhs.parallel(rhs)) {
