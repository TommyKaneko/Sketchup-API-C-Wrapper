52SUSceneRef Scene::copy_reference(const Scene& other) {
53 if (other.m_attached || SUIsInvalid(other.m_entity)) {
54 return other.ref();
55 }
56 SUSceneRef new_scene = create_scene();
57 // Copy basic properties from the source scene
58 SUSceneRef other_ref = other.ref();
59 // Copy name
60 SUStringRef name = SU_INVALID;
61 SUStringCreate(&name);
62 if (SUSceneGetName(other_ref, &name) == SU_ERROR_NONE) {
63 size_t len = 0;
64 SUStringGetUTF8Length(name, &len);
65 if (len > 0) {
66 std::string name_str(len, '\0');
67 size_t out_len = 0;
68 SUStringGetUTF8(name, len + 1, &name_str[0], &out_len);
69 SUSceneSetName(new_scene, name_str.c_str());
70 }
71 }
72 SUStringRelease(&name);
73 // Copy description
74 SUStringRef desc = SU_INVALID;
75 SUStringCreate(&desc);
76 if (SUSceneGetDescription(other_ref, &desc) == SU_ERROR_NONE) {
77 size_t len = 0;
78 SUStringGetUTF8Length(desc, &len);
79 if (len > 0) {
80 std::string desc_str(len, '\0');
81 size_t out_len = 0;
82 SUStringGetUTF8(desc, len + 1, &desc_str[0], &out_len);
83 SUSceneSetDescription(new_scene, desc_str.c_str());
84 }
85 }
86 SUStringRelease(&desc);
87 // Copy camera
88 SUCameraRef other_cam = SU_INVALID;
89 if (SUSceneGetCamera(other_ref, &other_cam) == SU_ERROR_NONE) {
90 SUSceneSetCamera(new_scene, other_cam);
91 }
92 // Copy flags
93 uint32_t flags = 0;
94 if (SUSceneGetFlags(other_ref, &flags) == SU_ERROR_NONE) {
95 SUSceneSetFlags(new_scene, flags);
96 }
97 return new_scene;
98}

481std::vector<Layer> Scene::layers() const {
482 if (!(*this)) {
483 throw std::logic_error("CW::Scene::layers(): Scene is null");
484 }
485 size_t count = 0;
486 SUResult res = SUSceneGetNumLayers(this->ref(), &count);
487 assert(res == SU_ERROR_NONE); _unused(res);
488 if (count == 0) {
489 return std::vector<Layer>();
490 }
491 std::vector<SULayerRef> layer_refs(count, SU_INVALID);
492 res = SUSceneGetLayers(this->ref(), count, layer_refs.data(), &count);
493 assert(res == SU_ERROR_NONE); _unused(res);
494 std::vector<Layer> result;
495 result.reserve(count);
496 for (size_t i = 0; i < count; ++i) {
497 result.emplace_back(layer_refs[i], true);
498 }
499 return result;
500}

503void Scene::add_layer(const Layer& layer) {
504 if (!(*this)) {
505 throw std::logic_error("CW::Scene::add_layer(): Scene is null");
506 }
507 SUResult res = SUSceneAddLayer(this->ref(), layer.ref());
508 switch (res) {
509 case SU_ERROR_NO_DATA: {
510 throw std::logic_error("CW::Scene::add_layer(): Scene is not owned by a valid model");
511 }
512 case SU_ERROR_GENERIC: {
513 throw std::logic_error("CW::Scene::add_layer(): the scene's model does not contain the layer");
514 }
515 case SU_ERROR_INVALID_ARGUMENT: {
516 throw std::invalid_argument("CW::Scene::add_layer(): the layer already exists in the scene");
517 }
518 default: {
519 assert(res == SU_ERROR_NONE); _unused(res);
520 }
521 }

567Scene Scene::copy() const {
568 if (!(*this)) {
569 throw std::logic_error("CW::Scene::copy(): Scene is null");
570 }
571 Scene new_scene(std::string(this->name()));
572 new_scene.description(this->description());
573 new_scene.flags(this->flags());
574 // Note: camera is not copied here because SUSceneSetCamera only works
575 // reliably on scenes that have been added to a model. Use
576 // scene.camera(source.camera()) after adding the scene to a model.
577 new_scene.copy_attributes_from(*this);
578 return new_scene;
579}
