45/**
46 * @brief C++ wrapper for SUEdgeUseRef.
47 *
48 * An EdgeUse represents the use of an edge in a specific loop/face context.
49 * It provides topology information for polygon boundaries. EdgeUse objects
50 * are obtained from loops and faces and are always read-only (no
51 * Create/Release).
52 *
53 * The start/end vertices of an EdgeUse may differ from the underlying edge's
54 * vertices because the loop may traverse the edge in reverse.
55 *
56 * @see SUEdgeUseRef
57 */
58class EdgeUse :public Entity {
59 private:
60 static SUEdgeUseRef copy_reference(const EdgeUse& other);
61
62 public:
63 /**
64 * @brief Constructs a NULL EdgeUse object.
65 */
66 EdgeUse();
67
68 /**
69 * @brief Constructs an EdgeUse from an existing SUEdgeUseRef.
70 * @param edge_use - the SUEdgeUseRef to wrap.
71 * @param attached - always true since EdgeUse objects are owned by the model.
72 */
73 EdgeUse(SUEdgeUseRef edge_use, bool attached = true);
74
75 /**
76 * @brief Copy constructor.
77 */
78 EdgeUse(const EdgeUse& other);
79
80 /**
81 * @brief Copy assignment operator.
82 */
83 EdgeUse& operator=(const EdgeUse& other);
84
85 /**
86 * @brief Returns the SUEdgeUseRef object.
87 */
88 SUEdgeUseRef ref() const;
89
90 /**
91 * @brief Returns true if this is an invalid/null edge use.
92 */
93 bool operator!() const;
94
95 /**
96 * @brief Retrieves the edge associated with this edge use.
97 * @throws std::logic_error if the edge use is null.
98 */
99 Edge edge() const;
100
101 /**
102 * @brief Retrieves the loop that this edge use belongs to.
103 * @throws std::logic_error if the edge use is null.
104 */
105 Loop loop() const;
106
107 /**
108 * @brief Retrieves the face that this edge use belongs to.
109 * @throws std::logic_error if the edge use is null.
110 */
111 Face face() const;
112
113 /**
114 * @brief Retrieves the partner edge uses (edge uses of the same edge
115 * from adjacent faces).
116 * @throws std::logic_error if the edge use is null.
117 */
118 std::vector<EdgeUse> partners() const;
119
120 /**
121 * @brief Retrieves whether this edge use is reversed relative to the
122 * underlying edge's direction.
123 * @throws std::logic_error if the edge use is null.
124 */
125 bool is_reversed() const;
126
127 /**
128 * @brief Retrieves the previous edge use in the loop.
129 * @throws std::logic_error if the edge use is null.
130 */
131 EdgeUse previous() const;
132
133 /**

135 * @throws std::logic_error if the edge use is null.
136 */
137 EdgeUse next() const;
138
139 /**
140 * @brief Retrieves the start vertex of this edge use.
141 * @note May differ from the underlying edge's start vertex if reversed.
142 * @throws std::logic_error if the edge use is null.
143 */
144 Vertex start_vertex() const;
145
146 /**
147 * @brief Retrieves the end vertex of this edge use.
148 * @note May differ from the underlying edge's end vertex if reversed.
149 * @throws std::logic_error if the edge use is null.
150 */
151 Vertex end_vertex() const;
152
153 /**
154 * @brief Retrieves the normal vector at the start vertex.
155 * @throws std::logic_error if the edge use is null.
156 */
157 Vector3D start_vertex_normal() const;
158
159 /**
160 * @brief Retrieves the normal vector at the end vertex.
161 * @throws std::logic_error if the edge use is null.
162 */
163 Vector3D end_vertex_normal() const;
164};
