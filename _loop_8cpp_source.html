73LoopInput Loop::loop_input() const {
74 if(!(*this)) {
75 throw std::logic_error("CW::Loop::loop_input(): Loop is null");
76 }
77 std::vector<Edge> edges = this->edges();
78 std::vector<InputEdgeProperties> edge_properties;
79 edge_properties.reserve(edges.size());
80 for (size_t i=0; i < edges.size(); i++) {
81 InputEdgeProperties edge_prop;
82 edge_prop.hidden = edges[i].hidden();
83 edge_prop.soft = edges[i].soft();
84 edge_prop.smooth = edges[i].smooth();
85 edge_prop.material = edges[i].material();
86 edge_prop.layer = edges[i].layer();
87 edge_properties.push_back(edge_prop);
88 }
89 return LoopInput(edge_properties);
90}

93std::vector<Edge> Loop::edges() const {
94 if(!(*this)) {
95 throw std::logic_error("CW::Loop::edges(): Loop is null");
96 }
97 size_t count = 0;
98 SUResult res = SULoopGetNumVertices(this->ref(), &count);
99 assert(res == SU_ERROR_NONE);
100 std::vector<SUEdgeRef> edge_refs(count, SU_INVALID);
101 res = SULoopGetEdges(this->ref(), count, edge_refs.data(), &count);
102 assert(res == SU_ERROR_NONE); _unused(res);
103 std::vector<Edge> edges(count);
104 std::transform(edge_refs.begin(), edge_refs.end(), edges.begin(),
105 [](const SUEdgeRef& value){
106 return Edge(value);
107 });
108 return edges;
109}

178PointLoopClassify Loop::classify_point(const std::vector<Point3D>& loop_points, const Point3D& test_point) {
179 if(loop_points.size() < 3) {
180 throw std::invalid_argument("CW::Loop::classify_point(): Fewer than 3 points given - not a valid loop.");
181 }
182 if(!test_point) {
183 throw std::invalid_argument("CW::Loop::classify_point(): Point3D given is null");
184 }
185 // http://www.geeksforgeeks.org/how-to-check-if-a-given-point-lies-inside-a-polygon/
186 // First check that the test point is on the plane
187 Plane3D loop_plane = Plane3D::plane_from_loop(loop_points);
188 if (!loop_plane) {
189 throw std::invalid_argument("CW::Loop::classify_point(): Points given does not form a valid loop.");
190 }
191 else if (!loop_plane.on_plane(test_point)) {
192 return PointLoopClassify::PointNotOnPlane;
193 }
194 // Now check if it is on the vertices
195 for (size_t i=0; i < loop_points.size(); i++) {
196 if (loop_points[i] == test_point) {
197 return PointLoopClassify::PointOnVertex;
198 }
199 }
200 // Now check if it is on the edges
201 for (size_t i=0; i < loop_points.size(); i++) {
202 Point3D next_point;
203 if (i == loop_points.size()-1) {
204 next_point = loop_points[0];
205 }
206 else {
207 next_point = loop_points[i+1];
208 }
209 if (Line3D::on_line_segment(loop_points[i], next_point, test_point)) {
210 return PointLoopClassify::PointOnEdge;
211 }
212 }
213 // Now check if it is inside or outside, given that we know that the point is (a) on the same plane of the loop, and (b) not on the edge, and (c) not on the vertices.
214 // From: http://www.geeksforgeeks.org/how-to-check-if-a-given-point-lies-inside-a-polygon/
215 // We draw a line from the point, and if it intersects with the loop an even number of times, then it is outside, if it intersects an odd number of times, it is inside.
216 // The line to draw can be any vector that is coplanar to the loop's plane - the simple one is get a vector of two points within the loop.
217 assert(loop_points[0] != loop_points[1]);
218 Vector3D ray = Vector3D(loop_points[1] - loop_points[0]).unit();
219 // Colinear line segments create problems for the algorith below, so we ignore them by creating a new list of points to calculate that exclude colinear edges.
220 struct LoopPoint {
221 Point3D point;
222 Vector3D vector_to_next;
223 };
224 std::vector<LoopPoint> red_loop_points;
225 red_loop_points.reserve(loop_points.size());
226 for (size_t i=0; i < loop_points.size(); i++) {
227 Point3D next_point;
228 if (i == loop_points.size()-1) {
229 next_point = loop_points[0];
230 } else {
231 next_point = loop_points[i+1];
232 }
233 Vector3D next_vector = next_point - loop_points[i];
234 if (next_vector.unit() != ray.unit() && next_vector.unit() != -ray.unit()) {
235 red_loop_points.push_back(LoopPoint{loop_points[i], next_vector});
236 }
237 }
238 // Now we have a list of points with colinear points removed, find intersections with the line segments
239 size_t num_intersections = 0;
240 for (size_t i=0; i < red_loop_points.size(); i++) {
241 Point3D intersection = Point3D::ray_line_intersection(red_loop_points[i].point, red_loop_points[i].vector_to_next, test_point, ray, false);
242 if (!!intersection) {
243 // There is a case where the ray intersection lies on a vertex of the loop. Whether this should be counted as an intersection of both lines connected to the loop or just once through depends on the orientation of the ray relative to the lines connected to that vertex
244 if (intersection == red_loop_points[i].point) {
245 // Get the directions of the lines connected to the vertex
246 LoopPoint prev_point;
247 if (i == 0) {
248 prev_point = red_loop_points[red_loop_points.size()-1];
249 }
250 else {
251 prev_point = red_loop_points[i-1];
252 }
253 Vector3D prev_cross = ray.cross(prev_point.vector_to_next);
254 Vector3D next_cross = ray.cross(red_loop_points[i].vector_to_next);
255 assert(fabs(next_cross.length()) > Vector3D::EPSILON);
256 if (prev_cross.unit() != next_cross.unit()) {
257 // The cross product direction is in opposite directions - ie the lines connected to the vertex are on the same side of the ray, therefore, count this as an additional intersection
258 num_intersections++;
259 }
260 }
261 else {
262 num_intersections++;
263 }
264 }
265 }
266 if (num_intersections % 2 == 0) {
267 // Even number of intersections
268 return PointLoopClassify::PointOutside;
269 }
270 else {
271 return PointLoopClassify::PointInside;
272 }
273}
