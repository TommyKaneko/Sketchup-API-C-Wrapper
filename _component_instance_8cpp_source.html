47SUComponentInstanceRef ComponentInstance::copy_reference(const ComponentInstance& other) {
48 if (other.m_attached) {
49 return other.ref();
50 }
51 // The other face has not been attached to the model, so copy its properties to a new object
52 ComponentInstance new_instance = other.definition().create_instance();
53 new_instance.transformation(other.transformation());
54 new_instance.name(other.name());
55 return new_instance.ref();
56}

92/** Copy assignment operator */
93ComponentInstance& ComponentInstance::operator=(const ComponentInstance& other) {
94 if (!m_attached && SUIsValid(m_entity)) {
95 SUComponentInstanceRef instance;
96 // Workaround for bug in Sketchup API which prevents an entity that is a group to be cast directly into ComponentInstance
97 if (SUEntityGetType(m_entity) == SURefType_Group) {
98 SUGroupRef group = SUGroupFromEntity(m_entity);
99 instance = SUGroupToComponentInstance(group);
100 }
101 else {
102 instance = SUComponentInstanceFromEntity(m_entity);
103 }
104 SUResult res = SUComponentInstanceRelease(&instance);
105 assert(res == SU_ERROR_NONE); _unused(res);
106 }
107 m_entity = SUComponentInstanceToEntity(copy_reference(other));
108 DrawingElement::operator=(other);
109 return *this;
110}

113SUComponentInstanceRef ComponentInstance::ref() const {
114 // TODO: Due to a bug in the C API, SUComponentInstanceFromEntity returns null if the reference object is a Group. The workaround is cast to group then to a ComponentInstance.
115 SURefType type = this->entity_type();
116 if (type == SURefType_Group) {
117 SUGroupRef group_ref = SUGroupFromEntity(m_entity);
118 return SUGroupToComponentInstance(group_ref);
119 }
120 return SUComponentInstanceFromEntity(m_entity);
121}

183DynamicComponentInfo ComponentInstance::create_dc_info() const {
184 if (!(*this)) {
185 throw std::logic_error("CW::ComponentInstance::create_dc_info(): ComponentInstance is null");
186 }
187 SUDynamicComponentInfoRef dc_info = SU_INVALID;
188 SUResult res = SUComponentInstanceCreateDCInfo(this->ref(), &dc_info);
189 if (res == SU_ERROR_NO_DATA) {
190 return DynamicComponentInfo(); // Not a dynamic component
191 }
192 assert(res == SU_ERROR_NONE); _unused(res);
193 return DynamicComponentInfo(dc_info);
