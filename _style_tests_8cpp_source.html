68TEST(Style, CopyConstructor) {
69 CW::Style style("Copy Style");
70 style.description("Copy desc");
71 CW::Style copy(style);
72 EXPECT_FALSE(!copy);
73 EXPECT_EQ(std::string(copy.name()), "Copy Style");
74 EXPECT_EQ(std::string(copy.description()), "Copy desc");
75 // Different ref pointers for detached copies
76 EXPECT_NE(style.ref().ptr, copy.ref().ptr);
77}

79TEST(Style, CopyAssignment) {
80 CW::Style style("Assign Style");
81 CW::Style copy;
82 copy = style;
83 EXPECT_FALSE(!copy);
84 EXPECT_EQ(std::string(copy.name()), "Assign Style");

87TEST(Style, PropertyGetSet) {
88 CW::Style style("Prop Style");
89 // Set background color
90 SUColor su_color = {255, 0, 0, 255};
91 CW::Color color_obj(su_color);
92 CW::TypedValue bg_color(color_obj);
93 style.set_property(SUStyleBackgroundColor, bg_color);
94 CW::TypedValue retrieved = style.get_property(SUStyleBackgroundColor);
95 CW::Color color = retrieved.color_value();
96 SUColor raw = color.ref();
97 EXPECT_EQ(raw.red, 255);
98}
