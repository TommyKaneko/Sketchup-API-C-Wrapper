84Material::Material(const Material& other):

86 //Entity(SUMaterialToEntity(copy_reference(other)))
87 Entity(other, SUMaterialToEntity(copy_reference(other))) // Because Material class owns the entity, we call the Entity constructor, not the copy constructor.
88{
89 // Copy across properties of the other material
90 if (!other.m_attached && SUIsValid(m_entity)) {
91 this->name(other.name());
92 this->type(other.type());

96 this->texture(texture_copy);

101Material& Material::operator=(const Material& other) {
102 if (!m_attached && SUIsValid(m_entity)) {
103 SUMaterialRef material = this->ref();
104 SUResult res = SUMaterialRelease(&material);
105 assert(res == SU_ERROR_NONE); _unused(res);
106 }
107 m_entity = SUMaterialToEntity(copy_reference(other));
108
109 if (!other.m_attached && SUIsValid(other.m_entity)) {
110 // The other material has not been attached to the model, so copy its properties to a new object
111 this->name(other.name());
112 this->type(other.type());
113 this->opacity(other.opacity());
114 this->color(other.color());
115 Texture texture_copy = other.texture();
116 this->texture(texture_copy);
117 }
118 Entity::operator=(other);
119 return *this;
120}

152Material Material::copy() const {
153 if(!(*this)) {
154 throw std::logic_error("CW::Material::copy(): Material is null");
155 }
156 Material new_material(create_material(), false);
157 new_material.name(this->name());
158 new_material.type(this->type());
159 new_material.opacity(this->opacity());
160 new_material.color(this->color());
161 Texture temp_texture = this->texture().copy(); // TODO: there is a problem with texture scaling here - API issue
162 new_material.texture(temp_texture);
163 new_material.copy_attributes_from((*this));
164 return new_material;

266Texture Material::texture() const {
267 if(!(*this)) {
268 return Texture(); // return a NULL texture if Material is NULL
269 }

271 SUResult res = SUMaterialGetTexture(this->ref(), &get_texture);

279void Material::texture(Texture& texture) {
280 if(!(*this)) {
281 throw std::logic_error("CW::Material::texture(): Material is null");
282 }
283 if (!texture) {
284 // no texture to apply
285 return;
286 }
287 if (texture.attached()) {
288 Texture texture_copy = texture.copy();
289 return this->texture(texture_copy);
290 }
291 SUTextureRef texture_ref = texture.ref();
292 SUResult res = SUMaterialSetTexture(this->ref(), texture_ref);
293 texture.attached(true); // Texture is now attached to the material
294 assert(res == SU_ERROR_NONE); _unused(res);
295}
