52SUTextRef Text::copy_reference(const Text& other) {
53 if (other.m_attached || SUIsInvalid(other.m_entity)) {
54 return other.ref();
55 }
56 SUTextRef new_text = create_text();
57
58 // Copy string
59 SUStringRef str = SU_INVALID;
60 SUStringCreate(&str);
61 SUResult res = SUTextGetString(other.ref(), &str);
62 assert(res == SU_ERROR_NONE); _unused(res);
63 size_t length = 0;
64 SUStringGetUTF8Length(str, &length);
65 std::string text_str(length, '\0');
66 SUStringGetUTF8(str, length + 1, &text_str[0], &length);
67 SUStringRelease(&str);
68 res = SUTextSetString(new_text, text_str.c_str());
69 assert(res == SU_ERROR_NONE); _unused(res);
70
71 // Copy leader type
72 SUTextLeaderType leader;
73 res = SUTextGetLeaderType(other.ref(), &leader);
74 assert(res == SU_ERROR_NONE); _unused(res);
75 res = SUTextSetLeaderType(new_text, leader);
76 assert(res == SU_ERROR_NONE); _unused(res);
77
78 // Copy arrow type
79 SUArrowType arrow;
80 res = SUTextGetArrowType(other.ref(), &arrow);
81 assert(res == SU_ERROR_NONE); _unused(res);
82 res = SUTextSetArrowType(new_text, arrow);
83 assert(res == SU_ERROR_NONE); _unused(res);
84
85 // Copy point (simple form - no instance path)
86 SUPoint3D point;
87 SUInstancePathRef path = SU_INVALID;
88 res = SUTextGetPoint(other.ref(), &point, &path);
89 if (res == SU_ERROR_NONE) {
90 // Set point without instance path for the copy
91 res = SUTextSetPoint(new_text, &point, SU_INVALID);
92 assert(res == SU_ERROR_NONE); _unused(res);
93 if (SUIsValid(path)) {
94 SUInstancePathRelease(&path);
95 }
96 }
97
98 // Copy leader vector
99 SUVector3D vector;
100 res = SUTextGetLeaderVector(other.ref(), &vector);
101 if (res == SU_ERROR_NONE) {
102 SUTextSetLeaderVector(new_text, &vector);
103 }
104
105 // Copy color
106 SUColor color;
107 res = SUTextGetColor(other.ref(), &color);
108 if (res == SU_ERROR_NONE) {
109 SUTextSetColor(new_text, &color);
110 }
111
112 return new_text;
113}

407Text Text::copy() const {
408 if (!(*this)) {
409 throw std::logic_error("CW::Text::copy(): Text is null");
410 }
411 // Create a fresh text and copy all properties manually
412 SUTextRef new_ref = create_text();
413
414 // Copy string
415 SUStringRef str = SU_INVALID;
416 SUStringCreate(&str);
417 SUResult res = SUTextGetString(this->ref(), &str);
418 assert(res == SU_ERROR_NONE); _unused(res);
419 size_t length = 0;
420 SUStringGetUTF8Length(str, &length);
421 std::string text_str(length, '\0');
422 SUStringGetUTF8(str, length + 1, &text_str[0], &length);
423 SUStringRelease(&str);
424 res = SUTextSetString(new_ref, text_str.c_str());
425 assert(res == SU_ERROR_NONE); _unused(res);
426
427 // Copy leader type
428 SUTextLeaderType leader;
429 res = SUTextGetLeaderType(this->ref(), &leader);
430 assert(res == SU_ERROR_NONE); _unused(res);
431 res = SUTextSetLeaderType(new_ref, leader);
432 assert(res == SU_ERROR_NONE); _unused(res);
433
434 // Copy arrow type
435 SUArrowType arrow;
436 res = SUTextGetArrowType(this->ref(), &arrow);
437 assert(res == SU_ERROR_NONE); _unused(res);
438 res = SUTextSetArrowType(new_ref, arrow);
439 assert(res == SU_ERROR_NONE); _unused(res);
440
441 // Copy point (simple form - no instance path for cross-model copy)
442 SUPoint3D pt;
443 SUInstancePathRef path = SU_INVALID;
444 res = SUTextGetPoint(this->ref(), &pt, &path);
445 if (res == SU_ERROR_NONE) {
446 res = SUTextSetPoint(new_ref, &pt, SU_INVALID);
447 assert(res == SU_ERROR_NONE); _unused(res);
448 if (SUIsValid(path)) {
449 SUInstancePathRelease(&path);
450 }
451 }
452
453 // Copy leader vector
454 SUVector3D vec;
455 res = SUTextGetLeaderVector(this->ref(), &vec);
456 if (res == SU_ERROR_NONE) {
457 SUTextSetLeaderVector(new_ref, &vec);
458 }
459
460 // Copy color
461 SUColor col;
462 res = SUTextGetColor(this->ref(), &col);
463 if (res == SU_ERROR_NONE) {
464 SUTextSetColor(new_ref, &col);
465 }
466
467 // Copy font
468 SUFontRef font = SU_INVALID;
469 res = SUTextGetFont(this->ref(), &font);
470 if (res == SU_ERROR_NONE && SUIsValid(font)) {
471 SUTextSetFont(new_ref, font);
472 }
473
474 Text new_text(new_ref, false);
475 new_text.copy_attributes_from(*this);
476 return new_text;
477}
