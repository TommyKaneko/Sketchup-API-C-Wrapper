65// Model-based tests: copy scenes from source model to destination model
66class SceneModelTest : public ModelLoad {};
67
68
69TEST_F(SceneModelTest, CopyScenes) {
70 // Read scenes from the source model
71 std::vector<CW::Scene> source_scenes = m_model->scenes();
72 ASSERT_GT(source_scenes.size(), 0u);
73
74 // Copy each scene using copy() and add to the destination model
75 for (const auto& src : source_scenes) {
76 CW::Scene dest_scene = src.copy();
77 m_model_copy->add_scene(dest_scene);
78 // Camera must be set after adding scene to model
79 dest_scene.camera(src.camera());
80 }
81
82 // Read back from destination and verify properties match
83 std::vector<CW::Scene> copied_scenes = m_model_copy->scenes();
84 ASSERT_EQ(copied_scenes.size(), source_scenes.size());
85
86 for (size_t i = 0; i < source_scenes.size(); ++i) {
87 EXPECT_EQ(std::string(source_scenes[i].name()), std::string(copied_scenes[i].name()));
88 EXPECT_EQ(std::string(source_scenes[i].description()), std::string(copied_scenes[i].description()));
89 EXPECT_EQ(source_scenes[i].use_camera(), copied_scenes[i].use_camera());
90 EXPECT_EQ(source_scenes[i].include_in_animation(), copied_scenes[i].include_in_animation());
91
92 CW::Camera src_cam = source_scenes[i].camera();
93 CW::Camera dest_cam = copied_scenes[i].camera();
94 EXPECT_NEAR(src_cam.eye().x, dest_cam.eye().x, 1e-6);
95 EXPECT_NEAR(src_cam.eye().y, dest_cam.eye().y, 1e-6);
96 EXPECT_NEAR(src_cam.eye().z, dest_cam.eye().z, 1e-6);
97 EXPECT_NEAR(src_cam.target().x, dest_cam.target().x, 1e-6);
98 EXPECT_NEAR(src_cam.target().y, dest_cam.target().y, 1e-6);
99 EXPECT_NEAR(src_cam.target().z, dest_cam.target().z, 1e-6);
100 }
101
102 SaveModel("Scenes");
103}
