44SUImageRef Image::create_image(const std::string& file_path) {
45 SUImageRef image = SU_INVALID;
46 SUResult res = SUImageCreateFromFile(&image, file_path.c_str());
47 switch (res) {
48 case SU_ERROR_NONE:
49 break;
50 case SU_ERROR_NULL_POINTER_INPUT:
51 case SU_ERROR_SERIALIZATION:
52 throw std::invalid_argument("CW::Image::create_image(): Invalid file path or unable to read file: " + file_path);
53 default:
54 assert(res == SU_ERROR_NONE); _unused(res);
55 }
56 return image;
57}

75SUImageRef Image::copy_reference(const Image& other) {
76 if (other.m_attached || SUIsInvalid(other.m_entity)) {
77 return other.ref();
78 }
79 // Create a new image from the original's image rep and copy the transform
80 SUImageRepRef image_rep = SU_INVALID;
81 SUResult res = SUImageGetImageRep(other.ref(), &image_rep);
82 assert(res == SU_ERROR_NONE); _unused(res);
83
84 SUImageRef new_image = SU_INVALID;
85 res = SUImageCreateFromImageRep(&new_image, image_rep);
86 assert(res == SU_ERROR_NONE); _unused(res);
87 res = SUImageRepRelease(&image_rep);
88 assert(res == SU_ERROR_NONE); _unused(res);
89
90 // Copy transform
91 SUTransformation transform;
92 res = SUImageGetTransform(other.ref(), &transform);
93 assert(res == SU_ERROR_NONE); _unused(res);
94 res = SUImageSetTransform(new_image, &transform);
95 assert(res == SU_ERROR_NONE); _unused(res);
96
97 return new_image;
98}

230Image Image::copy() const {
231 if (!(*this)) {
232 throw std::logic_error("CW::Image::copy(): Image is null");
233 }
234 // Create a new image from the original's image rep
235 SUImageRepRef rep = SU_INVALID;
236 SUResult res = SUImageGetImageRep(this->ref(), &rep);
237 assert(res == SU_ERROR_NONE); _unused(res);
238
239 SUImageRef new_ref = SU_INVALID;
240 res = SUImageCreateFromImageRep(&new_ref, rep);
241 assert(res == SU_ERROR_NONE); _unused(res);
242 res = SUImageRepRelease(&rep);
243 assert(res == SU_ERROR_NONE); _unused(res);
244
245 // Copy transform
246 SUTransformation transform;
247 res = SUImageGetTransform(this->ref(), &transform);
248 assert(res == SU_ERROR_NONE); _unused(res);
249 res = SUImageSetTransform(new_ref, &transform);
250 assert(res == SU_ERROR_NONE); _unused(res);
251
252 Image new_image(new_ref, false);
253 new_image.copy_attributes_from(*this);
254 return new_image;
255}
