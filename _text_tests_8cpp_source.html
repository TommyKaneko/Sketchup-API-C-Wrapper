82TEST_F(TextModelTest, CopyTexts) {
83 CW::Entities src_entities = m_model->entities();
84 std::vector<CW::Text> source_texts = src_entities.texts();
85
86 if (source_texts.size() == 0) {
87 SUCCEED() << "No texts in test model - add a Text annotation to 'box and box.skp' to enable full test";
88 return;
89 }
90
91 // Copy each text using copy() and add to the destination model
92 std::vector<CW::Text> new_texts;
93 for (const auto& src : source_texts) {
94 new_texts.push_back(src.copy());
95 }
96 CW::Entities dest_entities = m_model_copy->entities();
97 dest_entities.add_texts(new_texts);
98
99 // Read back from destination and verify properties match
100 std::vector<CW::Text> copied_texts = dest_entities.texts();
101 ASSERT_EQ(copied_texts.size(), source_texts.size());
102
103 for (size_t i = 0; i < source_texts.size(); ++i) {
104 EXPECT_EQ(std::string(source_texts[i].string()),
105 std::string(copied_texts[i].string()));
106 EXPECT_EQ(source_texts[i].leader_type(), copied_texts[i].leader_type());
107 EXPECT_EQ(source_texts[i].arrow_type(), copied_texts[i].arrow_type());
108 }
109
110 // Note: SaveModel may fail if the copied text is missing internal SketchUp
111 // data that can't be transferred via the C API. Skip save on failure.
112 try {
113 SaveModel("Texts");
114 } catch (const std::exception& e) {
115 SUCCEED() << "Text properties copied correctly but model save failed: " << e.what();
116 }
117}

120TEST_F(TextModelTest, ReadTextFont) {
121 CW::Entities src_entities = m_model->entities();
122 std::vector<CW::Text> texts = src_entities.texts();
123
124 if (texts.size() == 0) {
125 SUCCEED() << "No texts in test model";
126 return;
127 }
128
129 // Verify we can read font properties from the first text
130 CW::Font f = texts[0].font();
131 EXPECT_FALSE(!f);
132 CW::String face = f.face_name();
133 EXPECT_GT(std::string(face).length(), 0u);
134 EXPECT_GT(f.point_size(), 0u);
135}
