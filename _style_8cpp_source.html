51SUStyleRef Style::copy_reference(const Style& other) {
52 if (other.m_attached || SUIsInvalid(other.m_entity)) {
53 return other.ref();
54 }
55 // For detached styles, create a new empty style and copy properties
56 SUStyleRef new_style = create_style();
57 // Copy name
58 SUStringRef name_ref = SU_INVALID;
59 SUStringCreate(&name_ref);
60 SUStyleGetName(other.ref(), &name_ref);
61 size_t length = 0;
62 SUStringGetUTF8Length(name_ref, &length);
63 std::string name(length, '\0');
64 SUStringGetUTF8(name_ref, length + 1, &name[0], &length);
65 SUStringRelease(&name_ref);
66 SUStyleSetName(new_style, name.c_str());
67 // Copy description
68 SUStringRef desc_ref = SU_INVALID;
69 SUStringCreate(&desc_ref);
70 SUStyleGetDescription(other.ref(), &desc_ref);
71 length = 0;
72 SUStringGetUTF8Length(desc_ref, &length);
73 std::string desc(length, '\0');
74 SUStringGetUTF8(desc_ref, length + 1, &desc[0], &length);
75 SUStringRelease(&desc_ref);
76 SUStyleSetDescription(new_style, desc.c_str());
77 return new_style;
78}

108 SUStyleRef style = this->ref();
109 SUResult res = SUStyleRelease(&style);
110 assert(res == SU_ERROR_NONE); _unused(res);

115Style& Style::operator=(const Style& other) {
116 if (!m_attached && SUIsValid(m_entity)) {
117 SUStyleRef style = this->ref();
118 SUResult res = SUStyleRelease(&style);
119 assert(res == SU_ERROR_NONE); _unused(res);
120 }
121 m_entity = SUStyleToEntity(copy_reference(other));
122 Entity::operator=(other);
123 return *this;
124}
