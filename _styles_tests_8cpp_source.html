49// Test Styles from model
50TEST_F(ModelLoad, StylesFromModel) {
51 using namespace CW;
52 ASSERT_FALSE(!m_model);
53 Styles model_styles = m_model->styles();
54 EXPECT_FALSE(!model_styles);
55 size_t count = model_styles.size();
56 // Every model has at least one style
57 EXPECT_GT(count, 0u);
58
59 // Verify we can get all styles
60 std::vector<Style> all_styles = model_styles.styles();
61 EXPECT_EQ(all_styles.size(), count);
62
63 // Verify active style exists
64 Style active = model_styles.active_style();
65 EXPECT_FALSE(!active);
66
67 // Verify selected style exists
68 Style selected = model_styles.selected_style();
69 EXPECT_FALSE(!selected);
70
71 // Verify each style has a name and guid
72 for (const auto& s : all_styles) {
73 EXPECT_FALSE(!s);
74 String name = s.name();
75 EXPECT_GT(std::string(name).size(), 0u);
76 String guid = s.guid();
77 EXPECT_GT(std::string(guid).size(), 0u);
78 }
79
80 // Verify lookup by guid
81 if (!all_styles.empty()) {
82 std::string guid = std::string(all_styles[0].guid());
83 Style found = model_styles.style_by_guid(guid);
84 EXPECT_FALSE(!found);
85 EXPECT_EQ(std::string(found.guid()), guid);
86 }
87
88 // active_style_changed should be queryable
89 bool changed = model_styles.active_style_changed();
90 // Just verify it doesn't throw
91 (void)changed;
92}
