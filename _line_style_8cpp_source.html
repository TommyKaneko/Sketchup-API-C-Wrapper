43SULineStyleRef LineStyle::copy_reference(const LineStyle& other) {
44 if (other.m_attached || SUIsInvalid(other.m_entity)) {
45 return other.ref();
46 }
47 // Create a copy with the same name
48 SULineStyleRef copy = SU_INVALID;
49 SUStringRef name_ref = SU_INVALID;
50 SUStringCreate(&name_ref);
51 SULineStyleGetName(other.ref(), &name_ref);
52 size_t length = 0;
53 SUStringGetUTF8Length(name_ref, &length);
54 std::string name(length, '\0');
55 SUStringGetUTF8(name_ref, length + 1, &name[0], &length);
56 SUStringRelease(&name_ref);
57 SUResult res = SULineStyleCreateCopy(other.ref(), name.c_str(), &copy);
58 assert(res == SU_ERROR_NONE); _unused(res);
59 return copy;
60}

184Color LineStyle::color(bool& has_color) const {
185 if (!(*this)) {
186 throw std::logic_error("CW::LineStyle::color(): LineStyle is null");
187 }
188 SUColor su_color;
189 SUResult res = SULineStyleGetColor(this->ref(), &su_color);
190 if (res == SU_ERROR_NO_DATA) {
191 has_color = false;
192 return Color();
193 }
194 assert(res == SU_ERROR_NONE); _unused(res);
195 has_color = true;
196 return Color(su_color);
