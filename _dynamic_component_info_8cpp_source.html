102std::vector<DynamicComponentAttribute> DynamicComponentInfo::attributes() const {
103 if (!(*this)) {
104 throw std::logic_error("CW::DynamicComponentInfo::attributes(): DynamicComponentInfo is null");
105 }
106 size_t count = 0;
107 SUResult res = SUDynamicComponentInfoGetNumDCAttributes(m_dc_info, &count);
108 assert(res == SU_ERROR_NONE);
109 if (count == 0) {
110 return std::vector<DynamicComponentAttribute>();
111 }
112 std::vector<SUDynamicComponentAttributeRef> refs(count, SU_INVALID);
113 res = SUDynamicComponentInfoGetDCAttributes(m_dc_info, count, refs.data(), &count);
114 assert(res == SU_ERROR_NONE); _unused(res);
115 std::vector<DynamicComponentAttribute> result;
116 result.reserve(count);
117 for (size_t i = 0; i < count; ++i) {
118 result.push_back(DynamicComponentAttribute(refs[i]));
119 }
120 return result;
121}
