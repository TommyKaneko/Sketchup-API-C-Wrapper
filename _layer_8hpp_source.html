39/**
40 * @brief C++ wrapper for SULayerRef.
41 *
42 * A layer (also known as a Tag in newer SketchUp versions) controls
43 * visibility of drawing elements.
44 * @see SULayerRef
45 */
46class Layer :public Entity {
47 private:
48 /** @brief Creates a new empty SULayerRef via the C API. */
49 static SULayerRef create_layer();
50
51 /**
52 * @brief Creates a SULayerRef derived from an existing Layer object.
53 * @param other Layer object to derive the new ref from.
54 * @return If the Layer is already attached to a model, its existing ref
55 * is returned. Otherwise a new SULayerRef is created (properties
56 * will not be copied in that case).
57 */
58 static SULayerRef copy_reference(const Layer& other);
59
60 public:
61 /** @brief Constructs a null Layer. */
62 Layer();
63
64 /**
65 * @brief Wraps an existing SULayerRef.
66 * @param layer_ref Existing SULayerRef to wrap.
67 * @param attached true if the ref is already owned by a model.
68 */
69 Layer(SULayerRef layer_ref, bool attached = true);
70
71 /** @brief Copy constructor. */
72 Layer(const Layer& other);
73
74 /** @brief Destructor — releases the layer if not attached. */
75 ~Layer();
76
77 /** @brief Copy assignment operator. */
78 Layer& operator=(const Layer& other);
79
80 /** @brief Returns the underlying SULayerRef. */
81 SULayerRef ref() const;
82
83 /** @brief Implicit conversion to SULayerRef. */
84 operator SULayerRef() const;
85
86 /** @brief Returns a pointer to the underlying SULayerRef. */
87 operator SULayerRef*();
88
89 /**
90 * @brief Checks whether the Layer is invalid (null).
91 * @return true if the underlying ref is invalid.
92 */
93 bool operator!() const;
94
95 /**
96 * @brief Returns a detached copy of the Layer.
97 * @return A new Layer object not attached to any model.
98 * @throws std::logic_error if the Layer is null.
99 */
100 Layer copy() const;
101
102 /**
103 * @brief Retrieves the name of the layer.
104 * @return The layer name as a String.
105 * @throws std::logic_error if the Layer is null.
106 */
107 String name() const;
108
109 /**
110 * @brief Sets the name of the layer.
111 * @param string The new name for the layer.
112 * @throws std::logic_error if the Layer is null.
113 */
114 void name(const String& string);
115
116 /**
117 * @brief Sets the name of the layer.
118 * @param string The new name for the layer.
119 * @throws std::logic_error if the Layer is null.
120 */
121 void name(const std::string& string);
122
123 /** @brief Hash function for use with unordered_map. */
124 friend std::hash<CW::Layer>;
125};
