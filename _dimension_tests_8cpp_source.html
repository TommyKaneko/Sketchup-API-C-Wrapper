51TEST(DimensionLinearTest, CreateWithPoints) {
52 CW::Point3D start(0, 0, 0);
53 CW::Point3D end(10, 0, 0);
54 CW::DimensionLinear dim(start, end, 2.0);
55 EXPECT_FALSE(!dim);
56
57 // Check start and end points
58 CW::Point3D s = dim.start_point();
59 CW::Point3D e = dim.end_point();
60 EXPECT_NEAR(s.x, 0.0, 0.001);
61 EXPECT_NEAR(e.x, 10.0, 0.001);
62
63 // Check type via base class
64 EXPECT_EQ(dim.type(), SUDimensionType_Linear);
65}
