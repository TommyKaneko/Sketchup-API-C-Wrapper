46/**
47 * @brief Entity object wrapper.
48 *
49 * This is the base class for all SketchUp entities. Entities are basically
50 * anything that can be contained in a model, including DrawingElements such
51 * as Edges, SectionPlanes, Groups, etc. and entities that relate to those
52 * DrawingElements, such as Loops, Layers, etc.
53 *
54 * The methods below are available on all subclasses. For example, an Edge's
55 * parent class is DrawingElement, and a DrawingElement's parent class is
56 * Entity. Therefore an Edge has all of the methods defined in DrawingElement
57 * and Entity.
58 */
59class Entity {
60
61 friend class RubyAPI;
62 friend class InstancePath;
63
64 protected:
65 /**
66 * @brief The C SUEntityRef that this class wraps.
67 */
68 SUEntityRef m_entity;
69
70 /**
71 * @brief Indicates whether the Entity has been attached to a model.
72 */
73 bool m_attached;
74
75 /**
76 * @brief Constructor representing a null object.
77 */
78 Entity();
79
80 /**
81 * @brief Creates a new Entity object.
82 * @param entity SUEntityRef from the SketchUp C API.
83 * @param attached Flag indicating whether the entity is attached to a
84 * model. Defaults to true.
85 */
86 Entity(SUEntityRef entity, bool attached = true);
87
88 /**
89 * @brief Copy constructor with an optional parameter for the entity
90 * reference.
91 *
92 * SUEntityRef objects cannot be created from this class, so the Ref object
93 * must be passed to this constructor from a derived class object.
94 * @param other Entity object from which properties will be copied.
95 * @param entity_ref SUEntityRef object to assign to the copied object.
96 */
97 Entity(const Entity& other, SUEntityRef entity_ref = SU_INVALID);

99 /**
100 * @brief Destructor.

105 virtual ~Entity();

107 /**
108 * @brief Copy assignment operator.

110 Entity& operator=(const Entity& other);

112 /**
113 * @brief Copies properties of the entity from another entity.
114 * @param entity The source entity to copy properties from.

117 bool copy_properties_from(const Entity& entity);

119 public:
120 /**
121 * @brief Returns a copy of the wrapped SUEntityRef.

123 * the SUEntityRef could become invalid.

125 operator SUEntityRef() const;
126
127 /**
128 * @brief Returns a pointer to the wrapped SUEntityRef.
129 * @note The Entity object still manages the lifetime of the reference, so

132 operator SUEntityRef*();
133
134 /**
135 * @brief Returns the wrapped SUEntityRef.
136 * @note The Entity object still manages the lifetime of the reference, so

141 /**
142 * @brief Returns true if the entity is attached to another object.
143 */
144 bool attached() const;
145
146 /**
147 * @brief Sets the attached flag for the entity.
148 * @param attach true to indicate the entity has been attached to a model,
149 * false to indicate it has not.
150 */
151 void attached(bool attach);
152
153 /**
154 * @brief Retrieves the attribute dictionaries attached to the entity.
155 * @return vector of AttributeDictionary objects associated with the entity.
156 * If no dictionaries are associated, an empty vector is returned.
157 */
158 std::vector<AttributeDictionary> attribute_dictionaries() const;
159
160 /**
161 * @brief Retrieves an attribute dictionary with the given name. If a
162 * dictionary with the given name does not exist, one is added to
163 * the entity.
164 * @param name The name of the AttributeDictionary.
165 * @return AttributeDictionary object with the given name.
166 */
167 AttributeDictionary attribute_dictionary(const std::string& name) const;
168
169 /**
170 * @brief Adds an attribute dictionary to the entity. The given dictionary
171 * must not belong to another entity.
172 * @param dict The AttributeDictionary to add.
173 * @throws std::invalid_argument if the dictionary's name is empty or
174 * reserved for internal use.
175 * @throws std::logic_error if a dictionary with the same name already
176 * exists on the entity.
177 */
178 void add_dictionary(AttributeDictionary& dict);
179
180 /**
181 * @brief Copies attributes from another Entity object to this one.
182 * @param entity Object to copy the attributes from.
183 * @return true on success, false on failure.
184 */
185 bool copy_attributes_from(const Entity& entity);
186
187 /**
188 * @brief Checks if the entity is valid.
189 * @note This function does not check if the entity has been deleted.

191 */

195 * @brief Returns true if this entity is not valid. Alias of !is_valid().
196 */
197 virtual bool operator!() const;

200 * @brief Retrieves the unique ID assigned to an entity.
201 * @return The entity ID.

205 /**

208 * @param dict_name The name of the attribute dictionary.
209 * @param key The attribute key.
210 * @param default_value Default TypedValue to return if no attribute is

212 * @return The TypedValue of the attribute, or the default value if not
213 * found.

215 TypedValue get_attribute(const std::string& dict_name, const std::string& key, const TypedValue& default_value = TypedValue()) const;
216
217 /**
218 * @brief Retrieves the value of an attribute in the given attribute
219 * dictionary.
220 * @param dict AttributeDictionary object to search.
221 * @param key The attribute key.
222 * @param default_value Default TypedValue to return if no attribute is
223 * found. Defaults to an empty TypedValue.
224 * @return The TypedValue of the attribute, or the default value if not
225 * found.
226 */
227 TypedValue get_attribute(const AttributeDictionary &dict, const std::string& key, const TypedValue& default_value = TypedValue()) const;
228
229 /**
230 * @brief Sets the value of an attribute in the given attribute dictionary.
231 * @param dict_name The name of the AttributeDictionary.
232 * @param key The attribute key.
233 * @param value The TypedValue to set.
234 * @return true on success, false on failure.
235 */
236 void set_attribute(const std::string& dict_name, const std::string& key, const TypedValue& value);
237
238 /**
239 * @brief Sets the value of an attribute in the given attribute dictionary.
240 * @param dict The AttributeDictionary object.
241 * @param key The attribute key.
242 * @param value The TypedValue to set.
243 * @return true on success, false on failure.
244 */
245 void set_attribute(AttributeDictionary& dict, const std::string& key, const TypedValue& value);
246
247 /**
248 * @brief Returns the concrete type of the entity.
249 * @return The SURefType of the entity.
250 */
251 enum SURefType entity_type() const;
252
253 /**
254 * @brief Retrieves the model object associated with the entity.
255 * @throws std::logic_error if the entity is not associated with a model.
256 * @return The Model object that holds this entity.
257 */
258 Model model() const;
259
260 /**
261 * @brief Retrieves the parent Entities object that contains the entity.
262 * @throws std::logic_error if the entity is not contained by an Entities
263 * object.
264 * @return The parent Entities object.
265 */
266 Entities parent() const;
267
268 /**
269 * @brief Retrieves the persistent ID of the entity.

271 * @return The persistent ID.
272 */
273 int64_t persistent_id() const;
