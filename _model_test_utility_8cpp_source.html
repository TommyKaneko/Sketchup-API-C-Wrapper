50CW::Point3D SumPoints(const std::vector<CW::Point3D>& points) {
51 double sum_x = 0.0;
52 double sum_y = 0.0;
53 double sum_z = 0.0;
54 for (const auto& point : points) {
55 sum_x += point.x;
56 sum_y += point.y;
57 sum_z += point.z;
58 }
59 return CW::Point3D(sum_x, sum_y, sum_z);
60}

62// Helper function to try to sort the order of a vector of faces for comparison
63void SortFacesByProperties(std::vector<CW::Face>& faces) {
64 std::sort(faces.begin(), faces.end(),
65 [](const CW::Face& a, const CW::Face& b) {
66 CW::Point3D sum_a = SumPoints(a.outer_loop().points());
67 CW::Point3D sum_b = SumPoints(b.outer_loop().points());
68 if (std::abs(sum_a.x - sum_b.x) > 1e-6) {
69 return sum_a.x < sum_b.x;
70 }
71 if (std::abs(sum_a.y - sum_b.y) > 1e-6) {
72 return sum_a.y < sum_b.y;
73 }
74 return sum_a.z < sum_b.z;
75 });
76}

121void ModelLoad::ModelEntitiesOwnership(const CW::Entities& entities, const CW::Model* model) {
122 std::vector<CW::Face> faces = entities.faces();
123 for (const CW::Face& face : faces) {
124 this->ModelElementOwnership(face, model);
125 }
126 std::vector<CW::Edge> edges = entities.edges();
127 for (const CW::Edge& edge : edges) {
128 this->ModelElementOwnership(edge, model);
129 }
130 std::vector<CW::ComponentInstance> instances = entities.instances();
131 for (const CW::ComponentInstance& instance : instances) {
132 this->ModelElementOwnership(instance, model);
133 }
134 std::vector<CW::Group> groups = entities.groups();
135 for (const CW::Group& group : groups) {
136 this->ModelElementOwnership(group, model);
137 }

141void ModelLoad::ModelElementOwnership(const CW::DrawingElement& element, const CW::Model* model) {
142 ASSERT_TRUE(element.is_valid());
143 this->ModelEntityOwnership(element, model);
144 // Recursively check child entities
145 switch (element.entity_type()) {
146 case SURefType_Group: {
147 SUGroupRef group_ref = SUGroupFromDrawingElement(element.ref());
148 CW::Group group(group_ref);
149 CW::Entities group_entities = group.entities();
150 ModelEntitiesOwnership(group_entities, model);
151 break;
152 }
153 case SURefType_ComponentInstance: {
154 SUComponentInstanceRef instance_ref = SUComponentInstanceFromDrawingElement(element.ref());
155 CW::ComponentInstance instance(instance_ref);
156 this->ModelElementOwnership(instance.definition(), model);
157 break;
158 }
159 case SURefType_ComponentDefinition: {
160 SUComponentDefinitionRef definition_ref = SUComponentDefinitionFromDrawingElement(element.ref());
161 CW::ComponentDefinition definition(definition_ref);
162 CW::Entities definition_entities = definition.entities();
163 ModelEntitiesOwnership(definition_entities, model);
164 break;
165 }
166 default:
167 break;
168 }
169}
