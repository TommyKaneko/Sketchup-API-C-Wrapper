52SULayerFolderRef LayerFolder::copy_reference(const LayerFolder& other) {
53 if (other.m_attached || SUIsInvalid(other.m_entity)) {
54 return other.ref();
55 }
56 // Create a new folder with the same name
57 SUStringRef name_ref = SU_INVALID;
58 SUStringCreate(&name_ref);
59 SULayerFolderGetName(other.ref(), &name_ref);
60 size_t length = 0;
61 SUStringGetUTF8Length(name_ref, &length);
62 std::string name(length, '\0');
63 SUStringGetUTF8(name_ref, length + 1, &name[0], &length);
64 SUStringRelease(&name_ref);
65 return create_layer_folder(name);
66}

200void LayerFolder::add_layer(Layer& layer) {
201 if (!(*this)) {
202 throw std::logic_error("CW::LayerFolder::add_layer(): LayerFolder is null");
203 }
204 // SU API fails silently if the layer is not attached to the model already.
205 if (layer.attached() == false) {
206 throw std::invalid_argument("CW::LayerFolder::add_layer(): layer must be attached to a model");
207 }
208 else if (layer.model() != this->model()) {
209 throw std::invalid_argument("CW::LayerFolder::add_layer(): layer must belong to the same model as the layer folder");
210 }
211 SUResult res = SULayerFolderAddLayer(this->ref(), layer.ref());
212 switch (res) {
213 case SU_ERROR_INVALID_INPUT: {
214 throw std::invalid_argument("CW::LayerFolder::add_layer(): layer is not a valid object");
215 }
216 case SU_ERROR_INVALID_ARGUMENT: {
217 throw std::invalid_argument("CW::LayerFolder::add_layer(): layer can not be added to, or is already a member of, layer_folder");
218 }
219 default: {
220 assert(res == SU_ERROR_NONE); _unused(res);
221 }
222 }
223}

235std::vector<Layer> LayerFolder::layers() const {
236 if (!(*this)) {
237 throw std::logic_error("CW::LayerFolder::layers(): LayerFolder is null");
238 }
239 size_t count = 0;
240 SUResult res = SULayerFolderGetNumLayers(this->ref(), &count);
241 assert(res == SU_ERROR_NONE);
242 if (count == 0) {
243 return std::vector<Layer>(0);
244 }
245 std::vector<SULayerRef> refs(count, SU_INVALID);
246 res = SULayerFolderGetLayers(this->ref(), count, refs.data(), &count);
247 assert(res == SU_ERROR_NONE); _unused(res);
248 std::vector<Layer> result;
249 result.reserve(count);
250 for (size_t i = 0; i < count; ++i) {
251 result.emplace_back(refs[i], true);
252 }
253 return result;
254}

257void LayerFolder::add_layer_folder(LayerFolder& folder) {
258 if (!(*this)) {
259 throw std::logic_error("CW::LayerFolder::add_layer_folder(): LayerFolder is null");
260 }
261 if (!folder) {
262 throw std::invalid_argument("CW::LayerFolder::add_layer_folder(): folder is null");
263 }

265 throw std::invalid_argument("CW::LayerFolder::add_layer_folder(): folder must be attached to a model before adding to LayerFolder");
266 }
267 else if (folder.model() != this->model()) {
268 throw std::invalid_argument("CW::LayerFolder::add_layer_folder(): folder must belong to the same model as the layer folder");
269 }
270 SUResult res = SULayerFolderAddLayerFolder(this->ref(), folder.ref());
271 assert(res == SU_ERROR_NONE); _unused(res);
272}

300 }
301 SULayerFolderRef parent = SU_INVALID;
