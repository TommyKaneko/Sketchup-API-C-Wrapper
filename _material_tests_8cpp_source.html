42// Material Referencing Tests that references of a material point to the same object
43TEST_F(ModelLoad, MaterialReferencing)
44{
45 using namespace CW;
46 ASSERT_FALSE(!m_model);
47 std::vector<Material> materials = m_model->materials();
48 // The above operation should refer to the same materials
49 for (const Material& mat : materials) {
50 CW::Material material_ref = mat;
51 // We should have the same reference
52 EXPECT_EQ(material_ref.attached(), true);
53 EXPECT_EQ(mat.attached(), true);
54 EXPECT_EQ(material_ref.ref().ptr, mat.ref().ptr);
55 }
56
57 // Now check that copies of copies also have the same properties
58 for (const Material& mat : materials) {
59 // Test a copy
60 CW::Material material_copy = mat.copy();
61 EXPECT_NE(material_copy.ref().ptr, mat.ref().ptr); // material references must differ
62 EXPECT_EQ(mat.attached(), true);
63 EXPECT_EQ(material_copy.attached(), false);
64 if (!!mat.texture()) {
65 EXPECT_NE(material_copy.texture().ref().ptr, mat.texture().ref().ptr); // texture references must differ
66 EXPECT_NE(material_copy.texture().image_rep().ref().ptr, mat.texture().image_rep().ref().ptr); // image_rep references must differ
67 }
68 // We should have the same properties
69 MaterialsAreEqual(mat, material_copy);
70
71 // Test copy of a copy:
72 CW::Material material_copy_copy = material_copy.copy();
73 EXPECT_NE(material_copy_copy.ref().ptr, material_copy.ref().ptr);
74 EXPECT_EQ(material_copy_copy.attached(), false);
75 MaterialsAreEqual(material_copy_copy, material_copy);
76 }
77}

81TEST_F(ModelLoad, MaterialImplicitCopying)
82{
83 using namespace CW;
84 ASSERT_FALSE(!m_model);
85 std::vector<Material> materials = m_model->materials();
86 // The above operation should refer to the same materials
87 m_model_copy->add_materials(materials);
88 std::vector<Material> copied_materials = m_model_copy->materials();
89 EXPECT_EQ(materials.size(), copied_materials.size());
90 for (size_t i = 0; i < materials.size(); ++i) {
91 const Material& original_mat = materials[i];
92 const Material& copied_mat = copied_materials[i];
93 // We should have different references
94 EXPECT_NE(original_mat.ref().ptr, copied_mat.ref().ptr);
95 EXPECT_TRUE(copied_mat.attached());
96 if (!!copied_mat.texture()) {
97 EXPECT_NE(copied_mat.texture().ref().ptr, original_mat.texture().ref().ptr); // texture references must differ
98 EXPECT_NE(copied_mat.texture().image_rep().ref().ptr, original_mat.texture().image_rep().ref().ptr); // image_rep references must differ
99 }
100 // But the properties should be the same
101 MaterialsAreEqual(original_mat, copied_mat);
102 }

106// Material Referencing Tests that references of a material point to the same object
107TEST_F(ModelLoad, MaterialExplicitCopying)
108{
109 using namespace CW;
110 ASSERT_FALSE(!m_model);
111 std::vector<Material> materials = m_model->materials();
112 // Check copies are new independent objects
113 std::vector<Material> materials_copy; materials_copy.reserve(materials.size());
114 for (const Material& mat : materials) {
115 CW::Material material_copy = mat.copy();
116 ASSERT_FALSE(material_copy.attached());
117 // We should have different references
118 EXPECT_NE(material_copy.ref().ptr, mat.ref().ptr);
119 // But the properties should be the same
120 MaterialsAreEqual(material_copy, mat);
121 materials_copy.push_back(material_copy);
122 }
123
124 // The above operation should refer to the same materials
125 m_model_copy->add_materials(materials_copy);
126 std::vector<Material> copied_materials = m_model_copy->materials();
127 EXPECT_EQ(materials.size(), copied_materials.size());
128 for (Material& original_mat : materials) {
129 const auto found_material = std::find_if(copied_materials.begin(), copied_materials.end(),
130 [original_mat](const Material& value2){ return original_mat.name() == value2.name();});
131 if (found_material != copied_materials.end()) {
132 const Material& copied_mat = *found_material;
133 // We should have different references
134 EXPECT_NE(original_mat.ref().ptr, copied_mat.ref().ptr);
135 if (!!copied_mat.texture()) {
136 EXPECT_NE(copied_mat.texture().ref().ptr, original_mat.texture().ref().ptr); // texture references must differ
137 EXPECT_NE(copied_mat.texture().image_rep().ref().ptr, original_mat.texture().image_rep().ref().ptr); // image_rep references must differ
138 }
139 // But the properties should be the same
140 MaterialsAreEqual(original_mat, copied_mat);
141 EXPECT_TRUE(copied_mat.attached());
142 }
143 }
