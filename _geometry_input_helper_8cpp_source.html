148void GeometryInputPlus::load_materials(const std::vector<Material>& materials) {
149 // Check to see if each material is in the model already
150 std::vector<Material> target_materials = m_target_model->materials();
151 std::vector<Material> materials_to_add; materials_to_add.reserve(materials.size());
152 for (const Material& material : materials) {
153 bool matched = false;
154 for (const Material& target_material : target_materials) {
155 if (material == target_material) {
156 // We can directly add the material to the map
157 m_material_dict.add_reference(material, target_material);
158 matched = true;
159 break;
160 }
161 else if (material.name() == target_material.name()) {
162 // match with name
163 m_material_dict.add_reference(material, target_material);
164 matched = true;
165 break;
166 }
167 }
168 if (!matched) {
169 // Create a new material in the target model
170 materials_to_add.push_back(material);
171 }
172 }
173 if (materials_to_add.size() == 0) return;
174 // Add the materials that do not exist in the model.
175 m_target_model->add_materials(materials_to_add, false);
176 target_materials = m_target_model->materials(); // Refresh the list of materials.
177 for (const Material& material : materials_to_add) {
178 for (const Material& target_material : target_materials) {
179 if (material == target_material) {
180 // We can directly add the material to the map
181 m_material_dict.add_reference(material, target_material);
182 break;
183 }
184 else if (material.name() == target_material.name()) {
185 // match with name
186 m_material_dict.add_reference(material, target_material);
187 break;
188 }
189 }
190 }
191 return;
192}

249size_t GeometryInputPlus::add_face(const Face &face, bool copy_material_layer) {
250 if(!(*this)) {
251 throw std::logic_error("CW::GeometryInputPlus::add_face(): GeometryInput is null");
252 }
253 if(!face) {
254 throw std::invalid_argument("CW::GeometryInputPlus::add_face(): Face argument is null");
255 }
256 LoopInput outer_loop_input;
257 // Add outer loop
258 std::vector<Point3D> outer_points = face.outer_loop().points();
259 if(outer_points.size() < 3) {
260 throw std::logic_error("CW::GeometryInputPlus::add_face(): face has less than 3 points in outer loop");
261 }
262 std::vector<Edge> outer_edges = face.outer_loop().edges();
263 size_t first_vertex_index;
264 for (size_t i=0; i < outer_points.size(); ++i) {
265 size_t v_index = this->add_vertex(outer_points[i]);
266 outer_loop_input.add_vertex_index(v_index);
267 if (i==0) first_vertex_index = v_index;
268 if (outer_edges[i].hidden()) {
269 outer_loop_input.set_edge_hidden(i, true);
270 }
271 if (outer_edges[i].smooth()) {
272 outer_loop_input.set_edge_smooth(i, true);
273 }
274 if (outer_edges[i].soft()) {
275 outer_loop_input.set_edge_soft(i, true);
276 }
277 if (!!outer_edges[i].material()) {
278 outer_loop_input.set_edge_material(i, m_material_dict.get_reference(outer_edges[i].material()));
279 }
280 if (!!outer_edges[i].layer()) {
281 outer_loop_input.set_edge_layer(i, m_layer_dict.get_reference(outer_edges[i].layer()));
282 }
283 }
284 outer_loop_input.add_vertex_index(first_vertex_index); // Close the loop TODO: not strictly necessary, and messes with the m_edge_num count...
285
286 size_t added_face_index = GeometryInput::add_face(outer_loop_input);
287 // Add inner loops
288 std::vector<Loop> inner_loops = face.inner_loops();
289 for (size_t i=0; i < inner_loops.size(); ++i) {
290 LoopInput inner_loop_input;
291 std::vector<Point3D> inner_points = inner_loops[i].points();
292 std::vector<Edge> inner_edges = inner_loops[i].edges();
293 for (size_t j=0; j < inner_points.size(); ++j) {
294 size_t v_index = this->add_vertex(inner_points[j]);
295 inner_loop_input.add_vertex_index(v_index);
296 if (j==0) first_vertex_index = v_index;
297 if (inner_edges[j].hidden()) {
298 inner_loop_input.set_edge_hidden(j, true);
299 }
300 if (inner_edges[j].smooth()) {
301 inner_loop_input.set_edge_smooth(j, true);
302 }
303 if (inner_edges[j].soft()) {
304 inner_loop_input.set_edge_soft(j, true);
305 }
306 if (!!inner_edges[j].material()) {
307 inner_loop_input.set_edge_material(j, m_material_dict.get_reference(inner_edges[j].material()));
308 }
309 if (!!inner_edges[j].layer()) {
310 inner_loop_input.set_edge_layer(j, m_layer_dict.get_reference(inner_edges[j].layer()));
311 }
312 }
313 inner_loop_input.add_vertex_index(first_vertex_index); // Close the loop TODO: not strictly necessary, and messes with the m_edge_num count...
314 this->face_add_inner_loop(added_face_index, inner_loop_input);
315 }
316 if (copy_material_layer) {
317 // Add layer
318 Layer face_layer = face.layer();
319 if (!!face_layer) {
320 this->GeometryInput::face_layer(added_face_index, m_layer_dict.get_reference(face_layer));
321 face_layer.attached(true);
322 }
323 #if SketchUpAPI_VERSION_MAJOR < 2021
324 // Old way of setting the material
325 Material front_mat = face.material();
326 if (!!front_mat) {
327 MaterialInput front_material_input(front_mat);
328 this->face_front_material(added_face_index, front_material_input);
329 }
330 Material back_mat = face.back_material();
331 if (!!back_mat) {
332 MaterialInput back_material_input(face.back_material());
333 assert(m_target_model->material_exists(back_material_input.material()));
334 this->face_back_material(added_face_index, back_material_input);
335 }
336 #else
337 // New way of setting the material with MaterialInputPosition
338 if (!!face.material()) {
339 MaterialPositionInput material_input = face.material_position_front();
340 // Replace material with one in the target model
341 material_input.material(this->material_reference(face.material()));
342 this->face_front_material_position(added_face_index, material_input);
343 }
344 if (!!face.back_material()) {
345 MaterialPositionInput material_input = face.material_position_back();
346 material_input.material(this->material_reference(face.back_material()));
347 this->face_back_material_position(added_face_index, material_input);
348 }
349 #endif
350 // TODO: create a way to add attributes to the faces.
351 }
352 return added_face_index;
353}

368size_t GeometryInputPlus::add_edge(const Edge &edge, const Material& material, const Layer& layer) {
369 if(!edge) {
370 throw std::invalid_argument("CW::GeometryInputPlus::add_edge(): Edge argument is null");
371 }
372 if (SU_API_VERSION_MAJOR < 5) {
373 // Edges can only be added since SU2017
374 return 0;
375 }
376 Point3D start_point = edge.start().position();
377 size_t start_index = this->add_vertex(start_point);
378 Point3D end_point = edge.end().position();
379 size_t end_index = this->add_vertex(end_point);
380 size_t added_edge_index = GeometryInput::add_edge(start_index, end_index);
381
382 // Add other information about the edge
383 if (edge.hidden()) {
384 this->edge_hidden(added_edge_index, true);
385 }
386 if (edge.smooth()) {
387 this->edge_smooth(added_edge_index, true);
388 }
389 if (edge.soft()) {
390 this->edge_soft(added_edge_index, true);
391 }
392 if (!!material) {
393 this->edge_material(added_edge_index, m_material_dict.get_reference(material));
394 }
395 if (!!layer) {
396 this->edge_layer(added_edge_index, m_layer_dict.get_reference(layer));
397 }
398 return added_edge_index;
399}

414size_t GeometryInputPlus::add_edges(const std::vector<Edge>& edges, const std::vector<Material>& materials, const std::vector<Layer>& layers) {
415 if(!(*this)) {
416 throw std::logic_error("CW::GeometryInputPlus::add_edges(): GeometryInput is null");
417 }
418 if(edges.size() != materials.size() || edges.size() != layers.size()) {
419 throw std::logic_error("CW::GeometryInputPlus::add_edges(): size of layers and materials vector must match the size of edges");
420 }
421 size_t edge_index = 0;
422 for (size_t i=0; i < edges.size(); ++i) {
423 edge_index = this->add_edge(edges[i], materials[i], layers[i]);
424 }
425 return edge_index;
426}

429void GeometryInputPlus::add_entities(const Entities& entities) {
430 // Add materials
431 std::vector<Material> materials = entities.model().materials();
432 this->load_materials(materials);
433 // Add layers
434 std::vector<Layer> layers = entities.model().layers();
435 this->load_layers(layers);
436 // Add faces
437 std::vector<Face> faces = entities.faces();
438 this->add_faces(faces, true);
439 // Add edges
440 std::vector<Edge> edges = entities.edges();
441 this->add_edges(edges);
442}
