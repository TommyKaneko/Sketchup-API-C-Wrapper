97bool DrawingElement::copy_properties_from(const DrawingElement& element) {
98 this->casts_shadows(element.casts_shadows());
99 this->hidden(element.hidden());
100 this->receives_shadows(element.receives_shadows());
101 // assigning layer and material properties may fail if the element is not attached to the model, so we only attempt it if both elements are attached
102 if (!element.attached() || !this->attached()) {
103 return true; // No-op for detached elements
104 }
105 // Only a layer or material that is attached to the same model can be assigned to the element
106 if (element.model() != this->model()) {
107 // Look for a layer with the same name
108 std::vector<Layer> layers = this->model().layers();
109 Layer matching_layer;
110 for (const Layer& layer : layers) {
111 if (layer.name() == element.layer().name()) {
112 matching_layer = layer;
113 break;
114 }
115 }
116 if (matching_layer.is_valid()) {
117 this->layer(matching_layer);
118 } else {
119 // No matching layer found, so we should create one in the model
120 Layer new_layer = element.layer().copy();
121 std::vector<Layer> new_layers = {new_layer};
122 this->model().add_layers(new_layers);
123 this->layer(new_layer);
124 }
125
126 // Look for a material with the same name
127 std::vector<Material> materials = this->model().materials();
128 Material matching_material;
129 for (const Material& material : materials) {
130 if (material.name() == element.material().name()) {
131 matching_material = material;
132 break;
133 }
134 }
135 if (matching_material.is_valid()) {
136 this->material(matching_material);
137 } else {
138 // No matching material found, so we should create one in the model
139 Material new_material = element.material().copy();
140 std::vector<Material> new_materials = {new_material};
141 this->model().add_materials(new_materials);
142 this->material(new_material);
143 }
144 }
145 return true;
146}

231void DrawingElement::material(const Material& material) {
232 if (SUIsInvalid(m_entity)) {
233 throw std::logic_error("CW::DrawingElement::material(): DrawingElement is null");
234 }
235 if (!material.is_valid()) {
236 return; // No-op for invalid/empty material
237 }
238 if (!material.attached()) {
239 throw std::logic_error("CW::DrawingElement::material(): only a material that is attached to a model can be assigned to a DrawingElement");
240 }
241 SUResult res = SUDrawingElementSetMaterial(this->ref(), material);
242 switch (res) {
243 case SU_ERROR_INVALID_ARGUMENT: {
244 throw std::invalid_argument("CW::DrawingElement::material(): the material is owned by a layer or image");
245 }
246 default: {
247 assert(res == SU_ERROR_NONE); _unused(res);
248 }
249 }
