207// Model-based tests: copy camera from source model to destination model
208class CameraModelTest : public ModelLoad {};
209
210
211TEST_F(CameraModelTest, CopyModelCamera) {
212 // Read camera from the source model and copy it
213 CW::Camera src_camera = m_model->camera();
214 EXPECT_FALSE(!src_camera);
215 CW::Camera camera_copy = src_camera.copy();
216
217 // Apply the copied camera to the destination model's camera
218 CW::Camera dest_camera = m_model_copy->camera();
219 dest_camera.set_orientation(camera_copy.eye(), camera_copy.target(), camera_copy.up());
220 dest_camera.perspective(camera_copy.perspective());
221 if (camera_copy.perspective()) {
222 dest_camera.fov(camera_copy.fov());
223 } else {
224 dest_camera.orthographic_height(camera_copy.orthographic_height());
225 }
226
227 // Verify the copy preserved the source properties
228 EXPECT_NEAR(dest_camera.eye().x, src_camera.eye().x, 1e-6);
229 EXPECT_NEAR(dest_camera.eye().y, src_camera.eye().y, 1e-6);
230 EXPECT_NEAR(dest_camera.eye().z, src_camera.eye().z, 1e-6);
231 EXPECT_NEAR(dest_camera.target().x, src_camera.target().x, 1e-6);
232 EXPECT_NEAR(dest_camera.target().y, src_camera.target().y, 1e-6);
233 EXPECT_NEAR(dest_camera.target().z, src_camera.target().z, 1e-6);
234 EXPECT_EQ(dest_camera.perspective(), src_camera.perspective());
235
236 SaveModel("Camera");
237}
