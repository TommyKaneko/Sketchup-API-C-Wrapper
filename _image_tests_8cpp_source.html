74// NOTE: The test model "box and box.skp" may not contain Image entities.
75// If so, the model-based copy test below will need the test model updated

77class ImageModelTest : public ModelLoad {};
78

80TEST_F(ImageModelTest, CopyImages) {
81 CW::Entities src_entities = m_model->entities();
82 std::vector<CW::Image> source_images = src_entities.images();
83
84 if (source_images.size() == 0) {
85 // No images in model - skip the copy test but verify we can read
86 SUCCEED() << "No images in test model - add an image to 'box and box.skp' to enable full test";
87 return;
88 }
89
90 // Copy each image using copy() and add to the destination model
91 CW::Entities dest_entities = m_model_copy->entities();
92 for (auto& src_img : source_images) {
93 CW::Image new_img = src_img.copy();
94 dest_entities.add_image(new_img);
95 }
96
97 // Read back from destination and verify properties match
98 std::vector<CW::Image> copied_images = dest_entities.images();
99 ASSERT_EQ(copied_images.size(), source_images.size());
100
101 for (size_t i = 0; i < source_images.size(); ++i) {
102 auto [src_w, src_h] = source_images[i].dimensions();
103 auto [dst_w, dst_h] = copied_images[i].dimensions();
104 EXPECT_DOUBLE_EQ(src_w, dst_w);
105 EXPECT_DOUBLE_EQ(src_h, dst_h);
106 }
107
108 SaveModel("Images");
109}
