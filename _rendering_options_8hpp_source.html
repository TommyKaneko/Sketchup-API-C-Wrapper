40/**
41 * @brief Wrapper for SURenderingOptionsRef, providing access to the
42 * rendering options of a SketchUp model.
43 *
44 * Rendering options control visual settings such as edge display,
45 * transparency, section cuts, and other render-time parameters.
46 * Options are accessed by key name and stored as TypedValue objects.
47 *
48 * @see https://extensions.sketchup.com/developer_center/sketchup_c_api/
49 */
50class RenderingOptions
51{
52 public:
53 /** @brief Default constructor – creates an invalid RenderingOptions. */
54 RenderingOptions();
55
56 /**
57 * @brief Constructs a RenderingOptions wrapping a raw API ref.
58 * @param ref the SURenderingOptionsRef to wrap.
59 */
60 RenderingOptions(SURenderingOptionsRef ref);
61
62 /** @brief Destructor. */
63 ~RenderingOptions();
64
65 /**
66 * @brief Returns all rendering option key names.
67 * @return a vector of key name strings.
68 */
69 std::vector<std::string> get_keys() const;
70
71 /**
72 * @brief Returns the value for the given rendering option key.
73 * @param key the key name to look up.
74 * @return the TypedValue associated with the key.
75 */
76 TypedValue get_value(const std::string& key) const;
77
78 /**
79 * @brief Sets the value for the given rendering option key.
80 * @param key the key name to set.
81 * @param tval the TypedValue to assign.
82 * @return true on success, false on failure.
83 */
84 bool set_value(const std::string& key, const TypedValue& tval);
85
86 private:
87 SURenderingOptionsRef m_rendering_options;
88
89}; // class RenderingOptions
