42SUArcCurveRef ArcCurve::create_arc_curve(
43 const Point3D& center, const Point3D& start_point,
44 const Point3D& end_point, const Vector3D& normal, size_t num_edges) {
45 SUArcCurveRef arc = SU_INVALID;
46 SUPoint3D c = center;
47 SUPoint3D s = start_point;
48 SUPoint3D e = end_point;
49 SUVector3D n = normal;
50 SUResult res = SUArcCurveCreate(&arc, &c, &s, &e, &n, num_edges);
51 assert(res == SU_ERROR_NONE); _unused(res);
52 return arc;
53}

56SUArcCurveRef ArcCurve::copy_reference(const ArcCurve& other) {
57 if (other.m_attached || SUIsInvalid(other.m_entity)) {
58 return other.ref();
59 }
60 // Create a new arc curve with the same geometry
61 SUPoint3D c, s, e;
62 SUVector3D n;
63 SUResult res = SUArcCurveGetCenter(other.ref(), &c);
64 assert(res == SU_ERROR_NONE);
65 res = SUArcCurveGetStartPoint(other.ref(), &s);
66 assert(res == SU_ERROR_NONE);
67 res = SUArcCurveGetEndPoint(other.ref(), &e);
68 assert(res == SU_ERROR_NONE);
69 res = SUArcCurveGetNormal(other.ref(), &n);
70 assert(res == SU_ERROR_NONE); _unused(res);
71 return create_arc_curve(Point3D(c), Point3D(s), Point3D(e), Vector3D(n),
72 other.num_edges());
73}
