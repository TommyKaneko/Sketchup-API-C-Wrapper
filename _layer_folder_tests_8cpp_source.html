117Layer find_layer_in_model_by_name(Model& model, const std::string& layer_name) {
118 std::vector<Layer> model_layers = model.layers();
119 for (Layer& model_layer : model_layers) {
120 if (std::string(model_layer.name()) == layer_name) {
121 return model_layer;
122 }
123 }
124 throw std::runtime_error("Copied layer not found in target model: " + layer_name);
125}

127void copy_layer_to_folder(const Layer& source_layer, LayerFolder& target_folder, Model& target_model) {
128 const std::string layer_name = std::string(source_layer.name());
129 std::vector<Layer> layers_to_add = {source_layer.copy()};
130 target_model.add_layers(layers_to_add);
131 Layer attached_layer = find_layer_in_model_by_name(target_model, layer_name);
132 target_folder.add_layer(attached_layer);
133}

136void add_layer_folders_nested(LayerFolder& parent_folder, const std::vector<LayerFolder>& layer_folder_list) {
137 using namespace CW;
138 Model target_model = parent_folder.model();
139 std::string parent_name = std::string(parent_folder.name());
140 for (const LayerFolder& folder : layer_folder_list) {
141 if (!!folder.parent_folder() && std::string(folder.parent_folder().name()) == parent_name) {
142 LayerFolder folder_copy = folder.copy();
143 parent_folder.model().add_layer_folder(folder_copy);
144 assert(folder_copy.attached());
145 parent_folder.add_layer_folder(folder_copy);
146 add_layer_folders_nested(folder_copy, folder.sub_folders());
147 // Also copy layers in the folder
148 std::vector<Layer> folder_layers = folder.layers();
149 for (const auto& layer : folder_layers) {
150 copy_layer_to_folder(layer, folder_copy, target_model);
151 }
152 }
153 }

157TEST_F(LayerFolderModelTest, CopyLayerFolderWithLayers) {
158 using namespace CW;
159 ASSERT_FALSE(!m_model_copy);
160
161 std::vector<LayerFolder> folders = m_model->layer_folders();
162 for (LayerFolder& folder: folders) {
163 if (!folder.parent_folder()) {
164 LayerFolder folder_copy = folder.copy();
165 m_model_copy->add_layer_folder(folder_copy);
166 EXPECT_TRUE(folder_copy.attached());
167 add_layer_folders_nested(folder_copy, folder.sub_folders());
168 // Also copy layers in the folder
169 std::vector<Layer> folder_layers = folder.layers();
170 for (const auto& layer : folder_layers) {
171 copy_layer_to_folder(layer, folder_copy, *m_model_copy);
172 }
173 }
174 }
175 std::vector<LayerFolder> copied_folders = m_model_copy->layer_folders();
176 EXPECT_EQ(copied_folders.size(), folders.size());
177 for (LayerFolder& source_folder : folders) {
178 // Find the corresponding source folder layer by name
179 bool found = false;
180 for (LayerFolder& copied_folder : copied_folders) {
181 if (std::string(source_folder.name()) == std::string(copied_folder.name())) {
182 found = true;
183 // Check equality of parent folder names (if any)
184 if (!!source_folder.parent_folder() && !!copied_folder.parent_folder()) {
185 EXPECT_EQ(std::string(source_folder.parent_folder().name()), std::string(copied_folder.parent_folder().name()));
186 } else {
187 EXPECT_TRUE(!source_folder.parent_folder());
188 EXPECT_TRUE(!copied_folder.parent_folder());
189 }
190 // Check equality of layer names in the folder
191 std::vector<Layer> source_layers = source_folder.layers();
192 std::vector<Layer> copied_layers = copied_folder.layers();
193 EXPECT_EQ(source_layers.size(), copied_layers.size()) << "Layer count mismatch in folder '" << std::string(source_folder.name()) << "'";
194 for (const Layer& source_layer : source_layers) {
195 // Layers may not be returned in the same order, so find the corresponding layer by name
196 bool layer_found = false;
197 for (const auto& cl : copied_layers) {
198 if (std::string(cl.name()) == std::string(source_layer.name())) {
199 layer_found = true;
200 break;
201 }
202 }
203 EXPECT_TRUE(layer_found) << "Layer '" << std::string(source_layer.name()) << "' not found in copied folder '" << std::string(copied_folder.name()) << "'";
204 }
205 }
206 }
207 EXPECT_TRUE(found);
208 }
209 SaveModel("LayerFolderWithLayers");
210}
