44// Test creating an arc (semicircle)
45TEST(ArcCurveTest, CreateArc)
46{
47 CW::Point3D center(0, 0, 0);
48 CW::Point3D start(10, 0, 0);
49 CW::Point3D end(-10, 0, 0);
50 CW::Vector3D normal(0, 0, 1);
51 size_t num_edges = 12;
52
53 CW::ArcCurve arc(center, start, end, normal, num_edges);
54 EXPECT_FALSE(!arc);
55
56 EXPECT_NEAR(arc.radius(), 10.0, 0.001);
57 EXPECT_FALSE(arc.is_full_circle());
58
59 CW::Point3D arc_center = arc.center();
60 EXPECT_NEAR(arc_center.x, 0.0, 0.001);
61 EXPECT_NEAR(arc_center.y, 0.0, 0.001);
62 EXPECT_NEAR(arc_center.z, 0.0, 0.001);
63
64 CW::Point3D arc_start = arc.start_point();
65 EXPECT_NEAR(arc_start.x, 10.0, 0.001);
66 EXPECT_NEAR(arc_start.y, 0.0, 0.001);
67 EXPECT_NEAR(arc_start.z, 0.0, 0.001);
68
69 CW::Point3D arc_end = arc.end_point();
70 EXPECT_NEAR(arc_end.x, -10.0, 0.001);
71 EXPECT_NEAR(arc_end.y, 0.0, 0.001);
72 EXPECT_NEAR(arc_end.z, 0.0, 0.001);
73
74 CW::Vector3D arc_normal = arc.normal();
75 EXPECT_NEAR(arc_normal.x, 0.0, 0.001);
76 EXPECT_NEAR(arc_normal.y, 0.0, 0.001);
77 EXPECT_NEAR(arc_normal.z, 1.0, 0.001);
78}

191// Test copy() creates a detached copy with the same geometry
192TEST(ArcCurveTest, Copy)
193{
194 CW::Point3D center(0, 0, 0);
195 CW::Point3D start(10, 0, 0);
196 CW::Point3D end(-10, 0, 0);
197 CW::Vector3D normal(0, 0, 1);
198 size_t num_edges = 16;
199
200 CW::ArcCurve arc(center, start, end, normal, num_edges);
201 CW::ArcCurve arc_copy = arc.copy();
202
203 EXPECT_FALSE(!arc_copy);
204 EXPECT_NEAR(arc_copy.radius(), arc.radius(), 0.001);
205 EXPECT_NEAR(arc_copy.center().x, center.x, 0.001);
206 EXPECT_NEAR(arc_copy.center().y, center.y, 0.001);
207 EXPECT_NEAR(arc_copy.center().z, center.z, 0.001);
208 EXPECT_NEAR(arc_copy.start_point().x, start.x, 0.001);
209 EXPECT_NEAR(arc_copy.start_point().y, start.y, 0.001);
210 EXPECT_NEAR(arc_copy.end_point().x, end.x, 0.001);
211 EXPECT_NEAR(arc_copy.end_point().y, end.y, 0.001);
212 EXPECT_NEAR(arc_copy.normal().z, normal.z, 0.001);
213 EXPECT_EQ(arc_copy.num_edges(), num_edges);
214}

246TEST_F(ArcCurveModelTest, CopyArcCurves) {
247 using namespace CW;
248 Entities src_entities = m_model->entities();
249 std::vector<ArcCurve> source_arcs = src_entities.arc_curves();
250
251 if (source_arcs.size() == 0) {
252 FAIL() << "No arc curves in test model - add arc curves to 'box and box.skp' to enable full test";
253 return;
254 }
255
256 // Use copy() to create detached copies for the destination model
257 std::vector<ArcCurve> new_arcs;
258 for (const auto& src : source_arcs) {
259 new_arcs.push_back(src.copy());
260 }
261
262 CW::Entities dest_entities = m_model_copy->entities();
263 dest_entities.add_arc_curves(new_arcs);
264
265 // Read back from destination and verify count matches
266 std::vector<ArcCurve> copied_arcs = dest_entities.arc_curves();
267 ASSERT_EQ(copied_arcs.size(), source_arcs.size());
268
269 SaveModel("ArcCurves");
270}
