68// Test EdgeUse from a face in the loaded model
69TEST_F(ModelLoad, EdgeUseFromFace)
70{
71 ASSERT_FALSE(!(*m_model));
72 CW::Entities entities = m_model->entities();
73 std::vector<CW::Face> faces = entities.faces();
74 ASSERT_GT(faces.size(), 0u);
75
76 // Get the outer loop of the first face
77 CW::Face& face = faces[0];
78 CW::Loop outer = face.outer_loop();
79 ASSERT_FALSE(!outer);
80
81 // Get edge uses from the loop via C API
82 size_t num_edges = outer.size();
83 ASSERT_GT(num_edges, 0u);
84
85 std::vector<SUEdgeUseRef> eu_refs(num_edges, SU_INVALID);
86 size_t count = 0;
87 SUResult res = SULoopGetEdgeUses(outer.ref(), num_edges, eu_refs.data(), &count);
88 ASSERT_EQ(res, SU_ERROR_NONE);
89 ASSERT_GT(count, 0u);
90
91 // Wrap in EdgeUse and test properties
92 CW::EdgeUse eu(eu_refs[0]);
93 EXPECT_FALSE(!eu);
94
95 // Test edge()
96 CW::Edge edge = eu.edge();
97 EXPECT_FALSE(!edge);
98
99 // Test loop()
100 CW::Loop eu_loop = eu.loop();
101 EXPECT_FALSE(!eu_loop);
102
103 // Test face()
104 CW::Face eu_face = eu.face();
105 EXPECT_FALSE(!eu_face);
106
107 // Test start and end vertex
108 CW::Vertex sv = eu.start_vertex();
109 EXPECT_FALSE(!sv);
110 CW::Vertex ev = eu.end_vertex();
111 EXPECT_FALSE(!ev);
112
113 // Test next/previous navigation (should form a cycle)
114 CW::EdgeUse next = eu.next();
115 EXPECT_FALSE(!next);
116 CW::EdgeUse prev = eu.previous();
117 EXPECT_FALSE(!prev);
118
119 // Test vertex normals
120 CW::Vector3D sn = eu.start_vertex_normal();
121 CW::Vector3D en = eu.end_vertex_normal();
122 // Normals should be non-zero
123 double sn_len = std::sqrt(sn.x * sn.x + sn.y * sn.y + sn.z * sn.z);
124 double en_len = std::sqrt(en.x * en.x + en.y * en.y + en.z * en.z);
125 EXPECT_GT(sn_len, 0.0);
126 EXPECT_GT(en_len, 0.0);
127}

129// Test EdgeUse partners (adjacent faces sharing an edge)
130TEST_F(ModelLoad, EdgeUsePartners)
131{
132 ASSERT_FALSE(!(*m_model));
133 CW::Entities entities = m_model->entities();
134 std::vector<CW::Face> faces = entities.faces();
135 ASSERT_GT(faces.size(), 0u);
136
137 // Find an edge use that has partners (shared edge between two faces)
138 for (const CW::Face& face : faces) {
139 CW::Loop outer = face.outer_loop();
140 size_t num_edges = outer.size();
141 std::vector<SUEdgeUseRef> eu_refs(num_edges, SU_INVALID);
142 size_t count = 0;
143 SUResult res = SULoopGetEdgeUses(outer.ref(), num_edges, eu_refs.data(), &count);
144 if (res != SU_ERROR_NONE) continue;
145
146 for (size_t i = 0; i < count; ++i) {
147 CW::EdgeUse eu(eu_refs[i]);
148 std::vector<CW::EdgeUse> partners = eu.partners();
149 if (partners.size() > 0) {
150 // Found one with partners — verify properties
151 EXPECT_FALSE(!partners[0]);
152 CW::Edge partner_edge = partners[0].edge();
153 EXPECT_FALSE(!partner_edge);
154 return; // Test passes
155 }
156 }
157 }
158 // If no partners found, that's OK for box model — edges are shared
159}
