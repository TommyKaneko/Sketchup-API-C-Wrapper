48SUTextureRef Texture::create_texture(ImageRep& image_rep) {
49 if(!image_rep) {
50 throw std::logic_error("CW::Texture::create_texture(): ImageRep is null");
51 }
52 SUImageRepRef image = SU_INVALID;
53 if (image_rep.m_attached) {
54 ImageRep image_rep_copy = image_rep.copy();
55 image = image_rep_copy.ref();
56 image_rep_copy.m_attached = true;
57 } else {
58 image = image_rep.ref();
59 image_rep.m_attached = true;
60 }
61 SUTextureRef texture = SU_INVALID;
62 SUResult res = SUTextureCreateFromImageRep(&texture, image);
63 assert(res == SU_ERROR_NONE); _unused(res);
64 return texture;
65}

67SUTextureRef Texture::create_texture(const std::string file_path, double s_scale, double t_scale) {
68 SUTextureRef texture = SU_INVALID;
69 SUResult res = SUTextureCreateFromFile(&texture, file_path.c_str(), s_scale, t_scale);
70 if (res != SU_ERROR_NONE) {
71 return SU_INVALID;
72 }
73 // TODO: At time of writing, SUTextureCreateFromFile is broken - it does not set the s_scale and t_scale values. In SU2025 and over, we can use SUTextureSetDimensions to set these values after creation.
74 #if (SketchUpAPI_VERSION_MAJOR >= 2025)
75 res = SUTextureSetDimensions(texture, s_scale, t_scale);
76 if (res != SU_ERROR_NONE) {
77 return SU_INVALID;
78 }
79 #endif
80 return texture;
81}

84SUTextureRef Texture::copy_reference(const Texture& other) {
85 if (!other) {

170Texture Texture::copy() const {
171 if(!(*this)) {
172 // Texture is null
173 return Texture();
174 }
175 // @developers_notes Due to the lack of SUTextureSetDimensions() function in the C API until SU 2025, API 13.0, we have to use a workaround to copy the texture including its scale values.
176#if (SketchUpAPI_VERSION_MAJOR >= 2025)
177 // From SU 2025, API 13.0 onwards we can use the efficient method below.
178 ImageRep image_rep = this->image_rep().copy();
179 Texture copied_texture(image_rep);
180 copied_texture.file_name(this->file_name());
181 // Texture::set_dimensions uses inverse of scale values - weird but true. SO we have to invert the values here.
182 double si_scale = 1.0/this->s_scale();
183 double ti_scale = 1.0/this->t_scale();
184 copied_texture.set_dimensions(si_scale, ti_scale);
185 return copied_texture;
186#else
187 // For earlier versions we have to use the workaround method below.
188 // First create the file.
189 std::string file_path = "/tmp/tmptexture";
190 SUResult res = this->save_original(file_path);
191 assert(res == SU_ERROR_NONE); _unused(res);
192 double s_scale = this->s_scale();
193 double t_scale = this->t_scale();
194 #if defined(DEBUG)
195 std::cout << "Texture::copy() s_scale: " << s_scale << " t_scale: " << t_scale << "\n";
196 #endif
197 Texture new_texture(file_path, s_scale, t_scale);
198 // Delete the temporary file
199 const char *file_path_c = file_path.c_str();
200 int ret = std::remove(file_path_c);
201 assert(ret == 0); _unused(ret);
202 // Reset the file name
203 new_texture.file_name(this->file_name());
204 return new_texture;
205#endif
206}
